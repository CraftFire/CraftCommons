222        public static String ask(String name, String defaultvalue) {
223            String line = null;
224            String data = null;
225            boolean valid = false;
226            try {
227                do {
228                    System.out.println(newline + name + ": ");
229                    BufferedReader br = new BufferedReader(new InputStreamReader(
230                            System.in));
231                    line = br.readLine();
232                    if (line != null && !line.isEmpty()) {
233                        data = line;
234                        valid = true;
235                    } else {
236                        data = defaultvalue;
237                        System.out.println(defaultvalue);
238                        valid = true;
239                    }
240                } while (!valid);
241            } catch (IOException ioe) {
242                System.out.println("IO exception = " + ioe);
243            }
244            return data;
245        }

247        public static void printResult(String name, boolean success, Object x) {
248            String s = success ? "+" : "-";
249            ++count;
250            ++gcount;
251            if (success) {
252                ++succeed;
253                ++gsucceed;
254            }
255            s += name;
256            s += ": ";
257            if (x != null) {
258                s += x.toString().replace("\u0000", "\u2400");
259            } else {
260                s += "Doesn't work";
261            }
262            System.out.println(s);
263        }
264    
265        public static void printResult(String name, Object x) {
266            printResult(name, x != null, x);
267        }

269        public static String arrayToString(Object[] array) {
270            String s = "[";
271            if (array == null) {
272                return null;
273            }
274            for (int i = 0; i < array.length; ++i) {
275                s += array[i].toString() + ", ";
276            }
277            s += "]";
278            return s;
279        }

282            String s = "[";

284                return null;
285            }
286            for (int i = 0; i < array.length; ++i) {
287                s += array[i] + ", ";

289            s += "]";
