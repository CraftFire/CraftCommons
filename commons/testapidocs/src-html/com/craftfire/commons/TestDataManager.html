048        @Test
049        public void testURL() {
050            DummyDataManager dmH2 = new DummyDataManager(DataType.H2, null, null);
051            dmH2.getLogger().setDebug(true);
052            assertFalse(dmH2.setURL());
053            assertNull(dmH2.getURL());
054            dmH2.setDirectory("./target/test/blah/meh/");
055            dmH2.setDatabase("db");
056            assertTrue(dmH2.setURL());
057            assertEquals("jdbc:h2:./target/test/blah/meh/db;AUTO_RECONNECT=TRUE", dmH2.getURL());
058            datamanager.setHost(null);
059            datamanager.setDatabase(null);
060            assertFalse(datamanager.setURL());
061            assertNull(datamanager.getURL());
062            datamanager.setHost("localhost");
063            datamanager.setDatabase("db");
064            assertTrue(datamanager.setURL());
065            assertEquals("jdbc:mysql://localhost/db?zeroDateTimeBehavior=convertToNull&jdbcCompliantTruncation=false&autoReconnect=true&characterEncoding=UTF-8&characterSetResults=UTF-8",
066                    datamanager.getURL());
067            assertFalse(datamanager.hasConnection());
068        }

154        @Test
155        public void testFieldGettersFail() {
156            String table = "tb";
157            String name = "afield";
158            assertFalse(datamanager.getBooleanField(table, name, "1"));
159            assertNull(datamanager.getBinaryField(table, name, "1"));
160            assertNull(datamanager.getBlobField(table, name, "1"));
161            assertNull(datamanager.getDateField(table, name, "1"));
162            assertEquals(0, datamanager.getDoubleField(table, name, "1"), 0);
163            assertEquals(0, datamanager.getIntegerField(table, name, "1"));
164            assertNull(datamanager.getStringField(table, name, "1"));
165            assertFalse(datamanager.getBooleanField("SELECT `bool` FROM `" + table + "`"));
166            assertNull(datamanager.getBinaryField("SELECT `bin` FROM `" + table + "`"));
167            assertNull(datamanager.getBlobField("SELECT `blob` FROM `" + table + "`"));
168            assertNull(datamanager.getDateField("SELECT `date` FROM `" + table + "`"));
169            assertEquals(0, datamanager.getDoubleField("SELECT `double` FROM `" + table + "`"), 0);
170            assertEquals(0, datamanager.getIntegerField("SELECT `int` FROM `" + table + "`"));
171            assertNull(datamanager.getStringField("SELECT `vchar` FROM `" + table + "`"));
172        }
