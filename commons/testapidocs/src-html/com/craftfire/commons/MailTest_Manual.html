020    package com.craftfire.commons;
021    
022    import java.io.BufferedReader;
023    import java.io.IOException;
024    import java.io.InputStreamReader;
025    
026    import javax.mail.MessagingException;
027    import javax.mail.internet.InternetAddress;
028    import javax.mail.internet.MimeMessage;
029    import javax.mail.internet.MimeMessage.RecipientType;
030    
031    import com.craftfire.commons.util.MailSender;
032    
033    public class MailTest_Manual {
034        static final String newline = System.getProperty("line.separator");
035    
036        /**
037         * @param args
038         * @throws MessagingException
039         */
040        public static void main(String[] args) throws MessagingException {
041            MailSender sender;
042            String from, to, host, user, pass;
043            int port;
044            boolean ssl;
045            from = ask("Sender email", "dev@craftfire.com");
046            host = ask("Host", "localhost");
047            port = Integer.parseInt(ask("Port", "25"));
048            to = ask("Target email", "");
049            sender = new MailSender(new InternetAddress(from), host, port);
050            user = ask("Username", "");
051            pass = ask("Passwword", "");
052            ssl = ask("Use SSL", "false").equals("true");
053            sender.setUsername(user);
054            sender.setPassword(pass);
055            sender.setAuth(true);
056            sender.setSSL(ssl);
057            MimeMessage msg = sender.createMessage();
058            msg.setSubject("Testing");
059            msg.setRecipients(RecipientType.TO, to);
060            msg.setText("This is CraftCommons mail sending test.");
061            sender.send(msg);
062        }
063    
064        public static String ask(String name, String defaultvalue) {
065            String line = null;
066            boolean valid = false;
067            try {
068                System.out.println(newline + name + ": ");
069                BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
070                line = br.readLine();
071                if (line != null && !line.isEmpty()) {
072                    return line;
073                } else {
074                    System.out.println(defaultvalue);
075                    return defaultvalue;
076                }
077            } catch (IOException ioe) {
078                System.out.println("IO exception = " + ioe);
079            }
080            return null;
081        }
082    }
