111        public void testParseNull() throws Exception {
112            String input = "N;";

136        }
137    
138        @Test
139        public void testParseArray() throws Exception {
140            String input = "a:1:{i:1;i:2;}";
141            SerializedPhpParser serializedPhpParser = new SerializedPhpParser(input);
142            Object result = serializedPhpParser.parse();
143            assertTrue(result instanceof Map);
144            assertEquals(1, ((Map) result).size());
145            assertEquals(2, ((Map) result).get(1));
146        }
147    
148        @Test
149        public void testParseObject() throws Exception {
150            String input = "O:8:\"TypeName\":1:{s:3:\"foo\";s:3:\"bar\";}";
151            SerializedPhpParser serializedPhpParser = new SerializedPhpParser(input);
152            Object result = serializedPhpParser.parse();
153            assertTrue(result instanceof SerializedPhpParser.PhpObject);
154            assertEquals(1,
155                    ((SerializedPhpParser.PhpObject) result).attributes.size());
156            assertEquals("bar",
157                    ((SerializedPhpParser.PhpObject) result).attributes.get("foo"));
158    
159        }
160    
161        private void assertPrimitive(String input, Object expected) {
162            assertEquals(expected, new SerializedPhpParser(input).parse());
163        }
