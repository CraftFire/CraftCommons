105            assertTrue(ipv4.isIPv4());

198        @Test
199        public void testIPv4HashAndEquals() {
200            IPAddress addr = IPAddress.valueOf("192.168.1.1");
201            IPv4Address addr1 = new IPv4Address(192, 168, 2, 1);
202            assertTrue(ipv4.equals(ipv4));
203            assertTrue(ipv4.hashCode() == ipv4.hashCode());
204            assertTrue(ipv4.equals(addr));
205            assertTrue(addr.equals(ipv4));
206            assertTrue(ipv4.hashCode() == addr.hashCode());
207            assertFalse(ipv4.equals(addr1));
208            assertFalse(addr1.equals(ipv4));
209            assertFalse(ipv4.equals(ipv6));
210            assertFalse(addr.equals(ipv6));
211            assertFalse(ipv6.equals(addr));
212            assertFalse(ipv4.equals(ipv4.toIPv6()));
213            assertFalse(ipv4.toIPv6().equals(ipv4));
214            Map<IPv4Address, String> map = new HashMap<IPv4Address, String>();
215            map.put(ipv4, "alice");
216            map.put(addr1, "bob");
217            assertEquals("alice", map.get(addr));
218        }

242        @Test
243        public void testIPHashMaps() {
244            IPAddress v4 = IPAddress.valueOf("192.168.1.1");
245            IPv4Address v41 = new IPv4Address(192, 168, 2, 1);
246            IPAddress v6 = IPAddress.valueOf("fe80::1015:c0a8:1");
247            IPv6Address v61 = new IPv6Address(0x2002, 0, 0, 0, 0, 0, 0xcdfd, 0xc7b6);
248            Map<IPAddress, String> map = new HashMap<IPAddress, String>();
249            map.put(ipv4, "alice");
250            map.put(v61, "bob");
251            map.put(v41, "charlie");
252            map.put(v6, "dan");
253            assertEquals("alice", map.get(v4));
254            assertEquals("dan", map.get(ipv6));
255        }
