051        public void testTypeDetect() throws SerialException, SQLException {
052            assertEquals(ValueType.NULL, ValueHolderBase.typeDetect(null));
053            assertEquals(ValueType.STRING, ValueHolderBase.typeDetect("test"));
054            assertEquals(ValueType.INTEGER, ValueHolderBase.typeDetect(9));
055            assertEquals(ValueType.INTEGER, ValueHolderBase.typeDetect(-1L));
056            assertEquals(ValueType.INTEGER, ValueHolderBase.typeDetect(BigInteger.TEN));
057            assertEquals(ValueType.REAL, ValueHolderBase.typeDetect(12.5));
058            assertEquals(ValueType.REAL, ValueHolderBase.typeDetect(10.5f));
059            assertEquals(ValueType.REAL, ValueHolderBase.typeDetect(BigDecimal.valueOf(2.55)));
060            assertEquals(ValueType.DATE, ValueHolderBase.typeDetect(new Date()));
061            assertEquals(ValueType.BLOB, ValueHolderBase.typeDetect(new SerialBlob(new byte[] { 0, 1, 2, 17, 113, -10 })));
062            assertEquals(ValueType.BINARY, ValueHolderBase.typeDetect(new byte[] { 0, 1, 2, 17, 113, -10 }));
063            assertEquals(ValueType.BOOLEAN, ValueHolderBase.typeDetect(true));
064            assertEquals(ValueType.UNKNOWN, ValueHolderBase.typeDetect(new ArrayList<Object>()));
065            assertEquals(ValueType.UNKNOWN, ValueHolderBase.typeDetect(new HashMap<String, Object>()));
066        }

096        @Test
097        public void testBinary() {
098            ValueHolderBase holder = new ValueHolderBase(ValueType.BINARY, new byte[] { 1, 77, -3 });
099            assertEquals(ValueType.BINARY, holder.getType());
100            assertArrayEquals(new byte[] { 1, 77, -3 }, (byte[]) holder.getValue());
101            assertEquals(new String(new byte[] { 1, 77, -3 }), holder.getString());
102            assertEquals(85501, holder.getInt());
103            assertEquals(85501, holder.getLong());
104            assertEquals(BigInteger.valueOf(85501), holder.getBigInt());
105            assertEquals(4.2243106785072420773640808359E-319, holder.getDouble(), 0);
106            assertEquals(1.19812419998236184381049803101E-40, holder.getFloat(), 0);
107            assertArrayEquals(new byte[] { 1, 77, -3 }, holder.getBytes());
108            assertTrue(holder.getBool());
109            assertFalse(holder.isNull());
110        }

112        @Test
113        public void testBlob() throws SerialException, SQLException {
114            ValueHolderBase holder = new ValueHolderBase(ValueType.BLOB, new SerialBlob(new byte[] { 0, 1, 77, -3, 8, -14, 5, -76, 102 }));
115            assertEquals(ValueType.BLOB, holder.getType());
116            assertArrayEquals(new byte[] { 0, 1, 77, -3, 8, -14, 5, -76, 102 }, ((Blob) holder.getValue()).getBytes(1, 9));
117            assertEquals(new String(new byte[] { 0, 1, 77, -3, 8, -14, 5, -76, 102 }), holder.getString());
118            assertEquals(-234507162, holder.getInt());
119            assertEquals(94009382106674278L, holder.getLong());
120            assertEquals(BigInteger.valueOf(94009382106674278L), holder.getBigInt());
121            assertEquals(2.18649212959259492544895369281E-302, holder.getDouble(), 0);
122            assertEquals(-2.64829405664964992612190650368E30, holder.getFloat(), 0);
123            assertArrayEquals(new byte[] { 0, 1, 77, -3, 8, -14, 5, -76, 102 }, holder.getBytes());
124            assertArrayEquals(new byte[] { 0, 1, 77, -3, 8, -14, 5, -76, 102 }, holder.getBlob().getBytes(1, 9));
125            assertTrue(holder.getBool());
126            assertFalse(holder.isNull());
127        }

255        @Test
256        public void testDateFormats() {
257            Date now = new Date();
258            Calendar cal1 = Calendar.getInstance();
259            cal1.setTime(now);
260            Calendar cal2 = Calendar.getInstance();
261    
262            ValueHolderBase holder = new ValueHolderBase(ValueType.STRING, DateFormat.getDateInstance().format(now));
263            cal2.setTime(holder.getDate());
264            assertEquals(cal1.get(Calendar.YEAR), cal2.get(Calendar.YEAR));
265            assertEquals(cal1.get(Calendar.DAY_OF_YEAR), cal2.get(Calendar.DAY_OF_YEAR));
266    
267            holder = new ValueHolderBase(ValueType.STRING, DateFormat.getDateTimeInstance().format(now));
268            cal2.setTime(holder.getDate());
269            assertEquals(cal1.get(Calendar.YEAR), cal2.get(Calendar.YEAR));
270            assertEquals(cal1.get(Calendar.DAY_OF_YEAR), cal2.get(Calendar.DAY_OF_YEAR));
271            assertEquals(cal1.get(Calendar.HOUR_OF_DAY), cal2.get(Calendar.HOUR_OF_DAY));
272            assertEquals(cal1.get(Calendar.MINUTE), cal2.get(Calendar.MINUTE));
273            assertEquals(cal1.get(Calendar.SECOND), cal2.get(Calendar.SECOND));
274    
275            holder = new ValueHolderBase(ValueType.STRING, DateFormat.getTimeInstance().format(now));
276            cal2.setTime(holder.getDate());
277            assertEquals(cal1.get(Calendar.HOUR_OF_DAY), cal2.get(Calendar.HOUR_OF_DAY));
278            assertEquals(cal1.get(Calendar.MINUTE), cal2.get(Calendar.MINUTE));
279            assertEquals(cal1.get(Calendar.SECOND), cal2.get(Calendar.SECOND));
280    
281            holder = new ValueHolderBase(ValueType.STRING, DateFormat.getInstance().format(now));
282            cal2.setTime(holder.getDate());
283            assertEquals(cal1.get(Calendar.YEAR), cal2.get(Calendar.YEAR));
284            assertEquals(cal1.get(Calendar.DAY_OF_YEAR), cal2.get(Calendar.DAY_OF_YEAR));
285            assertEquals(cal1.get(Calendar.HOUR_OF_DAY), cal2.get(Calendar.HOUR_OF_DAY));
286            assertEquals(cal1.get(Calendar.MINUTE), cal2.get(Calendar.MINUTE));
287        }

311        @Test
312        public void testDefaults() throws SerialException, SQLException {
313            ValueHolderBase holder = new ValueHolderBase(null);
314            Random rnd = new Random();
315            long randomLong = rnd.nextLong();
316            int randomInt = (int) randomLong;
317            double randomDouble = rnd.nextDouble();
318            float randomFloat = (float) randomDouble;
319            byte[] randomBytes = new byte[10];
320            rnd.nextBytes(randomBytes);
321            assertEquals("test" + randomInt, holder.getString("test" + randomInt));
322            assertEquals(randomInt, holder.getInt(randomInt));
323            assertEquals(randomLong, holder.getLong(randomLong));
324            assertEquals(BigInteger.valueOf(randomInt), holder.getBigInt(BigInteger.valueOf(randomInt)));
325            assertEquals(randomFloat, holder.getFloat(randomFloat), 0);
326            assertEquals(randomDouble, holder.getDouble(randomDouble), 0);
327            assertEquals(BigDecimal.valueOf(randomDouble), holder.getDecimal(BigDecimal.valueOf(randomDouble)));
328            assertArrayEquals(randomBytes, holder.getBytes(randomBytes));
329            Blob randomBlob = new SerialBlob(randomBytes);
330            assertSame(randomBlob, holder.getBlob(randomBlob));
331            assertTrue(holder.getBool(true));
332            assertFalse(holder.getBool(false));
333        }
