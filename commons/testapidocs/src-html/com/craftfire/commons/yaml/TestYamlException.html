020    package com.craftfire.commons.yaml;
021    
022    import static org.junit.Assert.assertEquals;
023    import static org.junit.Assert.assertNull;
024    import static org.junit.Assert.assertSame;
025    import static org.junit.Assert.assertTrue;
026    import static org.mockito.Mockito.mock;
027    
028    import org.junit.Test;
029    
030    public class TestYamlException {
031        private static Throwable cause = mock(Throwable.class);
032        private static String msg = "Hey ho!";
033        private static String path = "my.awesome.node";
034    
035    
036        @Test
037        public void testConstructor1() {
038            YamlException e = new YamlException();
039            assertNull(e.getPath());
040            assertNull(e.getMessage());
041            assertNull(e.getCause());
042        }
043    
044        @Test
045        public void testConstructor2() {
046            YamlException e = new YamlException(msg);
047            assertNull(e.getPath());
048            assertEquals(msg, e.getMessage());
049            assertNull(e.getCause());
050        }
051    
052        @Test
053        public void testConstructor3() {
054            YamlException e = new YamlException(msg, path);
055            assertEquals(path, e.getPath());
056            assertTrue(e.getMessage().contains(msg));
057            assertTrue(e.getMessage().contains(path));
058            assertNull(e.getCause());
059        }
060    
061        @Test
062        public void testConstructor4() {
063            YamlException e = new YamlException(cause);
064            assertNull(e.getPath());
065            assertEquals(cause.toString(), e.getMessage());
066            assertSame(cause, e.getCause());
067        }
068    
069        @Test
070        public void testConstructor5() {
071            YamlException e = new YamlException(msg, cause);
072            assertNull(e.getPath());
073            assertEquals(msg, e.getMessage());
074            assertSame(cause, e.getCause());
075        }
076    
077        @Test
078        public void testConstructor6() {
079            YamlException e = new YamlException(msg, path, cause);
080            assertEquals(path, e.getPath());
081            assertTrue(e.getMessage().contains(msg));
082            assertTrue(e.getMessage().contains(path));
083            assertSame(cause, e.getCause());
084        }
085    
086    }
