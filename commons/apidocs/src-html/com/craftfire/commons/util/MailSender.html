020    package com.craftfire.commons.util;
021    
022    import java.security.Security;
023    import java.util.Properties;
024    
025    import javax.mail.MessagingException;
026    import javax.mail.Session;
027    import javax.mail.internet.AddressException;
028    import javax.mail.internet.InternetAddress;
029    import javax.mail.internet.MimeMessage;
030    
031    import com.sun.mail.smtp.SMTPTransport;
032    
033    /**
034     * A helper class for mail sending.
035     */
036    public class MailSender {
037        private String host, username, password;
038        private InternetAddress senderEmail;
039        private int port = 0;
040        private boolean ssl = false;
041        private boolean auth = false;
042        private Session session;
043    
044        /**
045         * Creates a new MailSender with given sender e-mail address and SMTP server
046         * 
047         * @param  senderEmail      e-mail address to use as the sender address
048         * @param  host             hostname of the SMTP server
049         * @param  port             port of the SMTP server
050         * @throws AddressException if the sender e-mail isn't a valid e-mail address
051         */
052        public MailSender(String senderEmail, String host, int port) throws AddressException {
053            this(new InternetAddress(senderEmail), host, port);
054        }
055    
056        /**
057         * Creates a new MailSender with given sender e-mail address and SMTP server
058         * 
059         * @param  senderEmail      e-mail address to use as the sender address
060         * @param  host             hostname of the SMTP server
061         * @param  port             port of the SMTP server
062         */
063        public MailSender(InternetAddress senderEmail, String host, int port) {
064            this(senderEmail);
065            this.host = host;
066            this.port = port;
067        }
068    
069        protected MailSender(InternetAddress senderEmail) {
070            this.senderEmail = senderEmail;
071        }
072    
073        /**
074         * Returns the SMTP server hostname.
075         * 
076         * @return SMTP server hostname
077         */
078        public String getHost() {
079            return this.host;
080        }
081    
082        /**
083         * Sets the SMTP server hostname.
084         * 
085         * @param host  SMTP server hostname
086         */
087        public void setHost(String host) {
088            this.host = host;
089            this.session = null;
090        }
091    
092        /**
093         * Rerurns the SMTP server port.
094         * 
095         * @return SMTP server port
096         */
097        public int getPort() {
098            return this.port;
099        }
100    
101        /**
102         * Sets the SMTP server port.
103         * 
104         * @param port  SMTP server port
105         */
106        public void setPort(int port) {
107            this.port = port;
108            this.session = null;
109        }
110    
111        /**
112         * Returns the username to used in SMTP authentication
113         * 
114         * @return  SMTP username
115         */
116        public String getUsername() {
117            return this.username;
118        }
119    
120        /**
121         * Sets the username to use in SMTP authentication
122         * 
123         * @param username  SMTP username
124         */
125        public void setUsername(String username) {
126            this.username = username;
127            this.session = null;
128        }
129    
130        /**
131         * Returns the password used in SMTP authentication
132         * 
133         * @return SMTP password
134         */
135        public String getPassword() {
136            return this.password;
137        }
138    
139        /**
140         * Sets the password to use in SMTP authentication
141         * 
142         * @param password  SMTP password
143         */
144        public void setPassword(String password) {
145            this.password = password;
146            this.session = null;
147        }
148    
149        /**
150         * Returns the e-mail address used as a sender address
151         * 
152         * @return sender e-mail address
153         */
154        public String getSender() {
155            return this.senderEmail.toString();
156        }
157    
158        /**
159         * Sets the e-mail address to use as a sender address
160         * 
161         * @param senderEmail       sender e-mail address
162         * @throws AddressException if the address isn't a valid e-mail address
163         */
164        public void setSender(String senderEmail) throws AddressException {
165            this.senderEmail = new InternetAddress(senderEmail);
166        }
167    
168        /**
169         * Returns the e-mail address used as a sender address
170         * 
171         * @return sender e-mail address
172         */
173        public InternetAddress getSenderAddress() {
174            return this.senderEmail;
175        }
176    
177        /**
178         * Sets the e-mail address to use as a sender address
179         * 
180         * @param senderEmail       sender e-mail address
181         */
182        public void setSenderAddress(InternetAddress senderEmail) {
183            this.senderEmail = senderEmail;
184        }
185    
186        /**
187         * Checks if SMTP authorization is enabled
188         * 
189         * @return true if enabled, false if not
190         */
191        public boolean isAuth() {
192            return this.auth;
193        }
194    
195        /**
196         * Sets if this sender should use SMTP authorization
197         * 
198         * @param auth  true if should, false if not
199         */
200        public void setAuth(boolean auth) {
201            this.auth = auth;
202            this.session = null;
203        }
204    
205        /**
206         * Checks if SSL is enabled
207         * 
208         * @return true if enabled, false if not
209         */
210        public boolean isSSL() {
211            return this.ssl;
212        }
213    
214        /**
215         * Sets if this sender should use SSL
216         * 
217         * @param ssl  true if should, false if not
218         */
219        public void setSSL(boolean ssl) {
220            this.ssl = ssl;
221            this.session = null;
222        }
223    
224        /**
225         * Sends an e-mail message with SMTP settings configured in this sender.
226         * 
227         * @param message             the message to send
228         * @throws MessagingException if a messaging error occurred
229         */
230        public void send(MimeMessage message) throws MessagingException {
231            SMTPTransport t = (SMTPTransport) getSession().getTransport(this.ssl ? "smtps" : "smtp");
232            try {
233                if (this.auth) {
234                    t.connect(this.host, this.username, this.password);
235                } else {
236                    t.connect(this.host, null, null);
237                }
238                t.sendMessage(message, message.getAllRecipients());
239            } finally {
240                t.close();
241            }
242        }
243    
244        /**
245         * Returns a mail session with all the SMTP settings of this sender.
246         * 
247         * @return the mail session
248         */
249        public Session getSession() {
250            if (this.session == null) {
251                Properties props = System.getProperties();
252                props.setProperty("mail.smtp.port", String.valueOf(this.port));
253                if (this.ssl) {
254                    Security.addProvider(new com.sun.net.ssl.internal.ssl.Provider());
255                    props.setProperty("mail.smtp.socketFactory.class", "javax.net.ssl.SSLSocketFactory");
256                    props.setProperty("mail.smtp.socketFactory.fallback", "false");
257                    props.setProperty("mail.smtp.socketFactory.port", String.valueOf(this.port));
258                    props.setProperty("mail.smtps.host", this.host);
259                    props.setProperty("mail.smtps.auth", String.valueOf(this.auth));
260                } else {
261                    props.setProperty("mail.smtp.host", this.host);
262                    props.setProperty("mail.smtp.auth", String.valueOf(this.auth));
263                }
264                props.put("mail.smtps.quitwait", "false");
265    
266                this.session = Session.getInstance(props, null);
267            }
268            return this.session;
269        }
270    
271        /**
272         * Creates a new e-mail message with sender address and all the SMTP settings configured in this sender.
273         * 
274         * @return                    a new message
275         * @throws MessagingException if a messaging error occurred
276         */
277        public MimeMessage createMessage() throws MessagingException {
278            MimeMessage message = new MimeMessage(getSession());
279            message.setSender(this.senderEmail);
280            return message;
281        }
