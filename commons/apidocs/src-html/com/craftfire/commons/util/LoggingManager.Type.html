020    package com.craftfire.commons.util;
021    
022    import java.io.BufferedWriter;
023    import java.io.File;
024    import java.io.FileWriter;
025    import java.io.IOException;
026    import java.text.DateFormat;
027    import java.text.SimpleDateFormat;
028    import java.util.Date;
029    import java.util.Map;
030    import java.util.logging.Level;
031    import java.util.logging.Logger;
032    
033    public class LoggingManager {
034        private final Logger logger;
035        private String prefix, directory, format = "HH:mm:ss";
036        private boolean debug = false, logging = false, combined = true;
037        protected static final Level debugLevel = new Level("DEBUG", Level.INFO.intValue() + 1) {
038        };
039    
040        public LoggingManager(String logger, String prefix) {
041            this.logger = Logger.getLogger(logger);
042            this.prefix = prefix;
043        }
044    
045        public static enum Type {
046            info, error, debug, warning
047        }
048    
049        public Logger getLogger() {
050            return this.logger;
051        }
052    
053        public String getPrefix() {
054            return this.prefix;
055        }
056    
057        public void setPrefix(String prefix) {
058            this.prefix = prefix;
059        }
060    
061        public String getDirectory() {
062            return this.directory;
063        }
064    
065        public void setDirectory(String directory) {
066            this.directory = directory;
067            if (directory == null || directory.isEmpty()) {
068                this.logging = false;
069            }
070        }
071    
072        public boolean isDebug() {
073            return this.debug;
074        }
075    
076        public void setDebug(boolean debug) {
077            this.debug = debug;
078        }
079    
080        public boolean isCombinedLogging() {
081            return this.combined;
082        }
083    
084        public void setCombinedLogging(boolean combined) {
085            this.combined = combined;
086        }
087    
088        public boolean isLogging() {
089            return this.logging;
090        }
091    
092        public void setLogging(boolean logging) {
093            this.logging = (this.directory != null) && (!this.directory.isEmpty()) && logging;
094        }
095    
096        public String getFormat() {
097            return this.format;
098        }
099    
100        public void setFormat(String format) {
101            this.format = format;
102        }
103    
104        public void info(String line) {
105            info(line, true);
106        }
107    
108        public void info(String line, boolean logFile) {
109            this.logger.info(this.prefix + " " + line);
110            if (logFile) {
111                toFile(Type.info, line);
112            }
113        }
114    
115        public void warning(String line) {
116            warning(line, true);
117        }
118    
119        public void warning(String line, boolean logFile) {
120            this.logger.warning(this.prefix + " " + line);
121            if (logFile) {
122                toFile(Type.warning, line);
123            }
124        }
125    
126        public void severe(String line) {
127            severe(line, true);
128        }
129    
130        public void severe(String line, boolean logFile) {
131            this.logger.severe(this.prefix + " " + line);
132            if (logFile) {
133                toFile(Type.error, line);
134            }
135        }
136    
137        public void debug(String line) {
138            debug(line, true);
139        }
140    
141        public void debug(String line, boolean logFile) {
142            if (isDebug()) {
143                this.logger.log(debugLevel, this.prefix + " " + line);
144                if (logFile) {
145                    toFile(Type.debug, line);
146                }
147            }
148        }
149    
150        public void error(String error) {
151            severe(error);
152        }
153    
154        public void advancedWarning() {
155            warning(System.getProperty("line.separator")
156                    + "|-----------------------------------------------------------------------------|"
157                    + System.getProperty("line.separator")
158                    + "|---------------------------------- WARNING ----------------------------------|"
159                    + System.getProperty("line.separator")
160                    + "|-----------------------------------------------------------------------------|", false);
161        }
162    
163        public void stackTrace(final Throwable e) {
164            stackTrace(e, null);
165        }
166    
167        public void stackTrace(final Throwable e, Map<Integer, String> extra) {
168            advancedWarning();
169            warning("Class name: " + e.getStackTrace()[1].getClassName(), false);
170            warning("Error message: " + e.getMessage(), false);
171            warning("Error cause: " + e.getCause(), false);
172            warning("File name: " + e.getStackTrace()[1].getFileName(), false);
173            warning("Function name: " + e.getStackTrace()[1].getMethodName(), false);
174            warning("Error line: " + e.getStackTrace()[1].getLineNumber(), false);
175            if (isLogging()) {
176                DateFormat logFormat = new SimpleDateFormat(this.format);
177                Date date = new Date();
178                warning("Check log file: " + this.directory + "error" + File.separator
179                        + logFormat.format(date) + "-error.log", false);
180            } else {
181                warning("Enable logging in the config to get more information about the error.", false);
182            }
183    
184            logError("--------------------------- STACKTRACE ERROR ---------------------------");
185            logError("Class name: " + e.getStackTrace()[1].getClassName());
186            logError("Error message: " + e.getMessage());
187            logError("Error cause: " + e.getCause());
188            logError("File name: " + e.getStackTrace()[1].getFileName());
189            logError("Function name: " + e.getStackTrace()[1].getMethodName());
190            logError("Error line: " + e.getStackTrace()[1].getLineNumber());
191            if (extra != null) {
192                for (int id : extra.keySet()) {
193                    logError(extra.get(id));
194                }
195            }
196            logError("--------------------------- STACKTRACE START ---------------------------");
197            for (int i = 0; i < e.getStackTrace().length; i++) {
198                logError(e.getStackTrace()[i].toString());
199            }
200            logError("---------------------------- STACKTRACE END ----------------------------");
201        }
202    
203        public void logError(String error) {
204            toFile(Type.error, error);
205        }
206    
207        private void toFile(Type type, String line) {
208            if (this.logging) {
209                File data = new File(this.directory, "");
210                if (!data.exists()) {
211                    if (data.mkdir()) {
212                        debug("Created missing directory: " + this.directory);
213                    }
214                }
215                data = new File(this.directory + type.toString() + File.separator, "");
216                if (!data.exists()) {
217                    if (data.mkdir()) {
218                        debug("Created missing directory: " + this.directory + type.toString());
219                    }
220                }
221                if (isCombinedLogging()) {
222                    data = new File(this.directory + "combined" + File.separator, "");
223                    if (!data.exists()) {
224                        if (data.mkdir()) {
225                            debug("Created missing directory: " + this.directory + type.toString());
226                        }
227                    }
228                }
229                DateFormat logFormat = new SimpleDateFormat(this.format);
230                Date date = new Date();
231                if (isCombinedLogging()) {
232                    data = new File(this.directory + "combined" + File.separator
233                            + logFormat.format(date) + "-combined.log");
234                    if (!data.exists()) {
235                        try {
236                            if (!data.createNewFile()) {
237                                error("Failed creating file for logging: '" + data.getName() + "'.");
238                            }
239                        } catch (IOException e) {
240                            stackTrace(e);
241                        }
242                    }
243                }
244                data = new File(this.directory + type.toString() + File.separator
245                        + logFormat.format(date) + "-" + type.toString() + ".log");
246                if (!data.exists()) {
247                    try {
248                        if (!data.createNewFile()) {
249                            error("Failed creating file for logging: '" + data.getName() + "'.");
250                        }
251                    } catch (IOException e) {
252                        stackTrace(e);
253                    }
254                }
255                if (isCombinedLogging()) {
256                    try {
257                        DateFormat stringFormat = new SimpleDateFormat("yyyy/MM/dd HH:mm:ss");
258                        FileWriter writer = new FileWriter(this.directory
259                                + "combined" + File.separator + logFormat.format(date) + "-"
260                                + "combined" + ".log", true);
261                        BufferedWriter buffer = new BufferedWriter(writer);
262                        buffer.write(stringFormat.format(date) + " - " + type.toString().toUpperCase()
263                                + " - " + line + System.getProperty("line.separator"));
264                        buffer.close();
265                        writer.close();
266                    } catch (IOException e) {
267                        stackTrace(e);
268                    }
269                }
270                try {
271                    DateFormat stringFormat = new SimpleDateFormat("yyyy/MM/dd HH:mm:ss");
272                    FileWriter writer = new FileWriter(this.directory
273                            + type.toString() + File.separator + logFormat.format(date) + "-"
274                            + type.toString() + ".log", true);
275                    BufferedWriter buffer = new BufferedWriter(writer);
276                    buffer.write(stringFormat.format(date) + " - " + line + System.getProperty("line.separator"));
277                    buffer.close();
278                    writer.close();
279                } catch (IOException e) {
280                    stackTrace(e);
281                }
282            }
283        }
284    }
