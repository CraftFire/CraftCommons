020    package com.craftfire.commons.util;
021    
022    
023    /**
024     * A class representing a version range.
025     */
026    public class VersionRange {
027        private Version min, max;
028    
029        /**
030         * Creates a version range with given minimal version and maximal version in string format, with default separator (<code>.</code>) and radix (<code>10</code>).
031         * 
032         * @param minVersion  the minimal version of the range
033         * @param maxVersion  the maximal version of the range
034         * @see               Version#Version(String)
035         */
036        public VersionRange(String minVersion, String maxVersion) {
037            this.min = new Version(minVersion);
038            this.max = new Version(maxVersion);
039        }
040        
041        /**
042         * Creates a version range with given minimal version and maximal version in string format, with the given <code>separator</code> and default radix (<code>10</code>).
043         * 
044         * @param minVersion  the minimal version of the range
045         * @param maxVersion  the maximal version of the range
046         * @param separator   the separator to use
047         * @see               Version#Version(String, String)
048         */
049        public VersionRange(String minVersion, String maxVersion, String separator) {
050            this.min = new Version(minVersion, separator);
051            this.max = new Version(maxVersion, separator);
052        }
053        
054        /**
055         * Creates a version range with given minimal version and maximal version in string format, with the given {@link separator} and {@link radix}.
056         * 
057         * @param minVersion  the minimal version of the range
058         * @param maxVersion  the maximal version of the range
059         * @param separator   the separator to use
060         * @param radix       the radix to use
061         * 
062         * @see               Version#Version(String, String, int)
063         */
064        public VersionRange(String minVersion, String maxVersion, String separator, int radix) {
065            this.min = new Version(minVersion, separator, radix);
066            this.max = new Version(maxVersion, separator, radix);
067        }
068    
069        /**
070         * Creates a version range with given minimal version and maximal version.
071         * 
072         * @param minVersion  the minimal version of the range
073         * @param maxVersion  the maximal version of the range
074         */
075        public VersionRange(Version minVersion, Version maxVersion) {
076            this.min = minVersion;
077            this.max = maxVersion;
078        }
079    
080        /**
081         * Returns the minimal version of the range
082         * 
083         * @return the minimal version of the range
084         */
085        public Version getMin() {
086            return this.min;
087        }
088    
089        /**
090         * Returns the maximal version of the range
091         * 
092         * @return the maximal version of the range
093         */
094        public Version getMax() {
095            return this.max;
096        }
097    
098        /**
099         * Checks if the given <code>version</code> in string format is within the range (between minimal and maximal version).
100         * 
101         * @param version  the version to check
102         * @return         <code>true</code> if the version is within range, <code>false</code> if not
103         */
104        public boolean inVersionRange(String version) {
105            return this.inVersionRange(new Version(version));
106        }
107    
108        /**
109         * Checks if the given <code>version</code> is within the range (between minimal and maximal version).
110         * 
111         * @param version  the version to check
112         * @return         <code>true</code> if the version is within range, <code>false</code> if not
113         */
114        public boolean inVersionRange(Version version) {
115            return (this.min.compareTo(version) <= 0) && (this.max.compareTo(version) >= 0);
116        }
117    
118        /* (non-Javadoc)
119         * @see java.lang.Object#toString()
120         */
121        @Override
122        public String toString() {
123            return "[" + this.getMin() + " - " + this.getMax() + "]";
124        }
125    }
