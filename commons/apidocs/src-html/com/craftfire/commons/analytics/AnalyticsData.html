020    package com.craftfire.commons.analytics;
021    
022    import java.util.HashMap;
023    import java.util.Map;
024    
025    public class AnalyticsData {
026        private AnalyticsItem name, version;
027        private Map<String, AnalyticsItem> customData = new HashMap<String, AnalyticsItem>();
028    
029        public AnalyticsData(String name, String version) {
030            this.name = new AnalyticsItem("project_name", name);
031            this.version = new AnalyticsItem("project_version",version);
032        }
033    
034        public AnalyticsItem getName() {
035            return this.name;
036        }
037    
038        public void setName(String name) {
039            getName().setValue(name);
040        }
041    
042        public AnalyticsItem getVersion() {
043            return this.version;
044        }
045    
046        public void setVersion(String version) {
047            getVersion().setValue(version);
048        }
049    
050        public void add(String key, String value) {
051            this.customData.put(key.toLowerCase(), new AnalyticsItem(key.toLowerCase(), value));
052        }
053    
054        public Map<String, AnalyticsItem> getCustomData() {
055            return this.customData;
056        }
057    
058        public AnalyticsItem getCustom(String key) {
059            if (getCustomData().containsKey(key.toLowerCase())) {
060                return getCustomData().get(key.toLowerCase());
061            }
062            return null;
063        }
064    
065        public AnalyticsItem getOSName() {
066            return new AnalyticsItem("os_name", System.getProperty("os.name"));
067        }
068    
069        public AnalyticsItem getOSVersion() {
070            return new AnalyticsItem("os_version", System.getProperty("os.version"));
071        }
072    
073        public AnalyticsItem getOSArch() {
074            return new AnalyticsItem("os_arch", System.getProperty("os.arch"));
075        }
076    
077        public AnalyticsItem getTotalMemory() {
078            return new AnalyticsItem("ram_total", Long.toString(Runtime.getRuntime().totalMemory()));
079        }
080    
081        public AnalyticsItem getMaxMemory() {
082            return new AnalyticsItem("ram_max", Long.toString(Runtime.getRuntime().maxMemory()));
083        }
084    
085        public AnalyticsItem getJavaVersion() {
086            return new AnalyticsItem("java_version", System.getProperty("java.version"));
087        }
088    }
