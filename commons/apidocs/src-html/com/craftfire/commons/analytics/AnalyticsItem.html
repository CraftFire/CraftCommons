020    package com.craftfire.commons.analytics;
021    
022    import java.io.UnsupportedEncodingException;
023    import java.net.URLEncoder;
024    
025    /**
026     * Data item of the analytics.
027     */
028    public class AnalyticsItem {
029        private String key, value;
030    
031        /**
032         * Default constructor.
033         *
034         * @param key    the key of the data item
035         * @param value  the value of the data item
036         */
037        public AnalyticsItem(String key, String value) {
038            this.key = key;
039            this.value = value;
040        }
041    
042        /**
043         * Returns the key of the data item.
044         *
045         * @return key of the data item
046         */
047        public String getKey() {
048            return this.key;
049        }
050    
051        /**
052         * Returns the key of the data item encoded in UTF-8.
053         *
054         * @return key of the data item in UTF-8
055         */
056        public String getKeyUTF8() {
057            try {
058                return URLEncoder.encode(this.key, "UTF-8");
059            } catch (UnsupportedEncodingException ignore) {
060                return this.key;
061            }
062        }
063    
064        /**
065         * Sets the key of the data item.
066         *
067         * @param key  key of the data item
068         */
069        public void setKey(String key) {
070            this.key = key;
071        }
072    
073        /**
074         * Returns the value of the data item.
075         *
076         * @return value of the data item
077         */
078        public String getValue() {
079            return this.value;
080        }
081    
082        /**
083         * Returns the value of the data item encoded in UTF-8.
084         *
085         * @return value of the data item in UTF-8
086         */
087        public String getValueUTF8() {
088            try {
089                return URLEncoder.encode(this.value, "UTF-8");
090            } catch (UnsupportedEncodingException ignore) {
091                return this.value;
092            }
093        }
094    
095        /**
096         * Sets the value of the data item.
097         *
098         * @param value  value of the data item
099         */
100        public void setValue(String value) {
101            this.value = value;
102        }
103    }
