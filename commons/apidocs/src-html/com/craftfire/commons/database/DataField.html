036    public class DataField extends AbstractValueHolder {
037        private final String table /* , type */;
038        private final int sqltype, size;
039        private final ValueHolder holder;
040    
041        public DataField(int column, ResultSetMetaData metaData, Object data) throws SQLException {
042            this.size = metaData.getColumnDisplaySize(column);
043            ValueType vtype = sqlTypeParse(this.sqltype, this.size, data);
044            String name = metaData.getColumnLabel(column);
045            boolean unsigned = metaData.getColumnTypeName(column).contains("UNSIGNED");
046            this.holder = new ValueHolderBase(vtype, name, unsigned, data);
047            this.sqltype = metaData.getColumnType(column);
048            this.table = metaData.getTableName(column);
049        }
050    
051        public DataField(ValueType type, int size, Object value) {
052            this.holder = new ValueHolderBase(type, value);
053            this.sqltype = Types.NULL;
054            this.table = "";
055            this.size = size;
056        }
057    
058        public DataField(String name, String table, int size, boolean unsigned,
059                Object data) {
060            this.holder = new ValueHolderBase(name, unsigned, data);
061            this.sqltype = Types.NULL;
062            this.table = table;
063            this.size = size;
064        }
065    
066        public DataField(ValueType type, String name, String table, int size,
067                boolean unsigned, Object data) {
068            this.holder = new ValueHolderBase(type, name, unsigned, data);
069            this.sqltype = Types.NULL;
070            this.table = table;
071            this.size = size;
072        }
073    
074        public DataField(int column, ResultSet resultset) throws SQLException {
075            ResultSetMetaData metaData = resultset.getMetaData();
076            int size = metaData.getColumnDisplaySize(column);
077            Object data;
078            if (metaData.getColumnType(column) == Types.BLOB || metaData.getColumnType(column) == Types.LONGVARBINARY) {
079                data = resultset.getBlob(column);
080            } else if (metaData.getColumnType(column) == Types.CLOB) {
081                data = resultset.getString(column);
082            } else {
083                data = resultset.getObject(column);
084            }
085            this.sqltype = metaData.getColumnType(column);
086            boolean unsigned = metaData.getColumnTypeName(column).contains("UNSIGNED");
087            ValueType vtype = sqlTypeParse(this.sqltype, size, data, resultset.wasNull());
088            String name = metaData.getColumnLabel(column);
089            this.holder = new ValueHolderBase(vtype, name, unsigned, data);
090            this.table = metaData.getTableName(column);
091            this.size = size;
092        }
093    
094        private static ValueType sqlTypeParse(int sqltype, int size, Object data, boolean wasNull) {
095            if (wasNull) {
096                return ValueType.NULL;
097            }
098            return sqlTypeParse(sqltype, size, data);
099        }
100    
101        private static ValueType sqlTypeParse(int sqltype, int size, Object data) {
102            if (data == null) {
103                return ValueType.NULL;
104            }
105            switch (sqltype) {
106            case Types.CHAR:
107            case Types.VARCHAR:
108            case Types.LONGVARCHAR:
109            case Types.CLOB:
110                return ValueType.STRING;
111            case Types.BLOB:
112            case Types.LONGVARBINARY:
113                return ValueType.BLOB;
114            case Types.BOOLEAN:
115                return ValueType.BOOLEAN;
116            case Types.BIT:
117                if (size <= 1) {
118                    return ValueType.BOOLEAN;
119                }
120                /* falls through */
121            case Types.BINARY:
122            case Types.VARBINARY:
123                return ValueType.BINARY;
124            case Types.TINYINT:
125                if (size <= 1) {
126                    return ValueType.BOOLEAN;
127                }
128                /* falls through */
129            case Types.SMALLINT:
130            case Types.INTEGER:
131            case Types.BIGINT:
132                return ValueType.INTEGER;
133            case Types.FLOAT:
134            case Types.DOUBLE:
135            case Types.REAL:
136            case Types.DECIMAL:
137                return ValueType.REAL;
138            case Types.DATE:
139            case Types.TIMESTAMP:
140            case Types.TIME:
141                return ValueType.DATE;
142            case Types.NULL:
143                return ValueType.NULL;
144            default:
145                // TODO: DataManager.getLogManager().warning("Unknown SQL type: " + sqltype + " Field data:" + data.toString());
146                return ValueType.UNKNOWN;
147            }
148        }
