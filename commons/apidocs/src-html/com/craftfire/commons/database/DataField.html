042        public DataField(int column, ResultSetMetaData metaData, Object data) throws SQLException {
043            this.sqltype = metaData.getColumnType(column);
044            this.size = metaData.getColumnDisplaySize(column);
045            this.data = data;
046            this.ftype = sqlTypeParse(this.sqltype, this.size, data);
047            this.typeCheck();
048            this.name = metaData.getColumnLabel(column);
049            this.table = metaData.getTableName(column);
050            this.unsigned = metaData.getColumnTypeName(column).contains("UNSIGNED");
051        }

083        public DataField(FieldType type, String name, String table, int size,
084                boolean unsigned, Object data) {
085            this.sqltype = Types.NULL;
086            if (data == null) {
087                    this.ftype = FieldType.NULL;
088            } else {
089                    this.ftype = type;
090            }
091            this.name = name;
092            this.table = table;
093            this.size = size;
094            this.unsigned = unsigned;
095            this.data = data;
096            this.typeCheck();
097        }
098    
099        public DataField(int column, ResultSet resultset) throws SQLException {
100            ResultSetMetaData metaData = resultset.getMetaData();
101            this.name = metaData.getColumnLabel(column);
102            this.table = metaData.getTableName(column);
103            this.size = metaData.getColumnDisplaySize(column);
104            if (metaData.getColumnType(column) == Types.BLOB || metaData.getColumnType(column) == Types.LONGVARBINARY) {
105                this.data = resultset.getBlob(column);
106            } else if (metaData.getColumnType(column) == Types.CLOB) {
107                this.data = resultset.getString(column);
108            } else {
109                this.data = resultset.getObject(column);
110            }
111            this.sqltype = metaData.getColumnType(column);
112            this.unsigned = metaData.getColumnTypeName(column).contains("UNSIGNED");
113            this.ftype = sqlTypeParse(this.sqltype, this.size, this.data, resultset.wasNull());
114        }
115    
116        private void typeCheck() {
117            IllegalArgumentException e = new IllegalArgumentException("Data: "
118                    + this.data.toString() + " doesn't match the type: "
119                    + this.ftype.name());
120            if (this.ftype.equals(FieldType.STRING)) {
121                if (!(this.data instanceof String)) {
122                    throw e;
123                }
124            } else if (this.ftype.equals(FieldType.INTEGER) || this.ftype.equals(FieldType.REAL)) {
125                if (!(this.data instanceof Number)) {
126                    throw e;
127                }
128            } else if (this.ftype.equals(FieldType.DATE)) {
129                if (!(this.data instanceof Date)) {
130                    throw e;
131                }
132            } else if (this.ftype.equals(FieldType.BLOB)) {
133                if (!(this.data instanceof Blob)) {
134                    throw e;
135                }
136            } else if (this.ftype.equals(FieldType.BINARY)) {
137                if (!(this.data instanceof byte[])) {
138                    throw e;
139                }
140            } else if (this.ftype.equals(FieldType.BOOLEAN)) {
141                if (!(this.data instanceof Boolean)) {
142                    throw e;
143                }
144            }
145        }

147        private static FieldType typeDetect(Object data) {
148            if (data == null) {
149                return FieldType.NULL;
150            } else if (data instanceof String) {
151                return FieldType.STRING;
152            } else if (data instanceof Number) {
153                if (((Number) data).doubleValue() == ((Number) data).longValue()) {
154                    return FieldType.INTEGER;
155                } else {
156                    return FieldType.REAL;
157                }
158            } else if (data instanceof Date) {
159                return FieldType.DATE;
160            } else if (data instanceof Blob) {
161                return FieldType.BLOB;
162            } else if (data instanceof byte[]) {
163                return FieldType.BINARY;
164            } else if (data instanceof Boolean) {
165                return FieldType.BOOLEAN;
166            }
167            //TODO: DataManager.getLogManager().warning("Unknown data type: " + data.toString());
168            return FieldType.UNKNOWN;
169        }
170    
171        private static FieldType sqlTypeParse(int sqltype, int size, Object data, boolean wasNull) {
172            if (wasNull) {
173                    return FieldType.NULL;
174            }
175            return sqlTypeParse(sqltype, size, data);
176        }
177    
178        private static FieldType sqlTypeParse(int sqltype, int size, Object data) {
179            if (data == null) {
180                    return FieldType.NULL;
181            }
182            switch (sqltype) {
183            case Types.CHAR:
184            case Types.VARCHAR:
185            case Types.LONGVARCHAR:
186            case Types.CLOB:
187                return FieldType.STRING;
188            case Types.BLOB:
189            case Types.LONGVARBINARY:
190                return FieldType.BLOB;
191            case Types.BOOLEAN:
192                return FieldType.BOOLEAN;
193            case Types.BIT:
194                if (size <= 1) {
195                    return FieldType.BOOLEAN;
196                }
197                /* falls through */
198            case Types.BINARY:
199            case Types.VARBINARY:
200                return FieldType.BINARY;
201            case Types.TINYINT:
202                if (size <= 1) {
203                    return FieldType.BOOLEAN;
204                }
205                /* falls through */
206            case Types.SMALLINT:
207            case Types.INTEGER:
208            case Types.BIGINT:
209                return FieldType.INTEGER;
210            case Types.FLOAT:
211            case Types.DOUBLE:
212            case Types.REAL:
213            case Types.DECIMAL:
214                return FieldType.REAL;
215            case Types.DATE:
216            case Types.TIMESTAMP:
217            case Types.TIME:
218                return FieldType.DATE;
219            case Types.NULL:
220                return FieldType.NULL;
221            default:
222                //TODO: DataManager.getLogManager().warning("Unknown SQL type: " + sqltype + " Field data:" + data.toString());
223                return FieldType.UNKNOWN;
224            }
225        }

251        public String getString() {
252            if (this.getFieldType().equals(FieldType.STRING)) {
253                return (String) this.data;
254            } else if (this.getFieldType().equals(FieldType.BINARY)
255                    || this.getFieldType().equals(FieldType.BLOB)) {
256                return new String(this.getBytes());
257            }
258            return this.data.toString();

265        public long getLong() {
266            if (this.getFieldType().equals(FieldType.INTEGER)
267                    || this.getFieldType().equals(FieldType.REAL)) {
268                return ((Number) this.data).longValue();
269            } else if (this.getFieldType().equals(FieldType.BOOLEAN)) {
270                return ((Boolean) this.data).booleanValue() ? 1 : 0;
271            } else if (this.getFieldType().equals(FieldType.DATE)) {
272                return ((Date) this.data).getTime();
273            } else if (this.getFieldType().equals(FieldType.BINARY)
274                    || this.getFieldType().equals(FieldType.BLOB)) {
275                byte[] bytes = { 0, 0, 0, 0, 0, 0, 0, 0 };
276                byte[] bytes1 = this.getBytes();
277                if (bytes1.length >= 8) {
278                    System.arraycopy(bytes1, 0, bytes, 0, 8);
279                } else {
280                    System.arraycopy(bytes1, 0, bytes, 8 - bytes1.length,
281                            bytes1.length);
282                }
283                return ByteBuffer.wrap(bytes).getLong();
284            } else if (this.getFieldType().equals(FieldType.STRING)) {
285                try {
286                    return Long.parseLong((String) this.data);
287                } catch (NumberFormatException e) {
288                }
289                return new Double(this.getDouble()).longValue();
290            }
291            return 0;
292        }

294        public BigInteger getBigInt() {
295            try {
296                if (this.data instanceof BigInteger) {
297                    return (BigInteger) this.data;
298                } else if (this.getFieldType().equals(FieldType.BOOLEAN)) {
299                    return ((Boolean) this.data).booleanValue() ? BigInteger.ONE
300                            : BigInteger.ZERO;
301                } else if (this.getFieldType().equals(FieldType.BINARY)
302                        || this.getFieldType().equals(FieldType.BLOB)) {
303                    byte[] bytes = this.getBytes();
304                    return new BigInteger(bytes);
305                } else if (this.getFieldType().equals(FieldType.STRING)) {
306                    return new BigInteger(this.data.toString());
307                } else {
308                    long l = 0;
309                    if (this.getFieldType().equals(FieldType.INTEGER)
310                            || this.getFieldType().equals(FieldType.REAL)) {
311                        l = ((Number) this.data).longValue();
312                    } else if (this.getFieldType().equals(FieldType.DATE)) {
313                        l = ((Date) this.data).getTime();
314                    } else {
315                        return null;
316                    }
317                    return new BigInteger(String.valueOf(l));
318                }
319            } catch (NumberFormatException e) {
320            }
321            return null;
322        }

324        public double getDouble() {
325            if (this.getFieldType().equals(FieldType.INTEGER)
326                    || this.getFieldType().equals(FieldType.REAL)) {
327                return ((Number) this.data).doubleValue();
328            } else if (this.getFieldType().equals(FieldType.BOOLEAN)) {
329                return ((Boolean) this.data).booleanValue() ? 1 : 0;
330            } else if (this.getFieldType().equals(FieldType.DATE)) {
331                return new Long(((Date) this.data).getTime()).doubleValue();
332            } else if (this.getFieldType().equals(FieldType.BINARY)
333                    || this.getFieldType().equals(FieldType.BLOB)) {
334                byte[] bytes = { 0, 0, 0, 0, 0, 0, 0, 0 };
335                byte[] bytes1 = this.getBytes();
336                if (bytes1.length >= 8) {
337                    System.arraycopy(bytes1, 0, bytes, 0, 8);
338                } else {
339                    System.arraycopy(bytes1, 0, bytes, 8 - bytes1.length,
340                            bytes1.length);
341                }
342                return ByteBuffer.wrap(bytes).getLong();
343            } else if (this.getFieldType().equals(FieldType.STRING)) {
344                try {
345                    return Double.parseDouble((String) this.data);
346                } catch (NumberFormatException e) {
347                }
348                try {
349                    return Double.parseDouble(((String) this.data)
350                            .replace(',', '.'));
351                } catch (NumberFormatException e) {
352                    e.getCause();
353                }
354            }
355            return 0;
356        }

362        public BigDecimal getDecimal() {
363            try {
364                if (this.data instanceof BigDecimal) {
365                    return (BigDecimal) this.data;
366                } else if (this.getFieldType().equals(FieldType.BOOLEAN)) {
367                    return ((Boolean) this.data).booleanValue() ? BigDecimal.ONE
368                            : BigDecimal.ZERO;
369                } else if (this.getFieldType().equals(FieldType.STRING)) {
370                    return new BigDecimal(this.data.toString());
371                } else if (this.getFieldType().equals(FieldType.INTEGER)
372                        || this.getFieldType().equals(FieldType.REAL)) {
373                    return new BigDecimal(((Number) this.data).doubleValue());
374                } else if (this.getFieldType().equals(FieldType.DATE)) {
375                    return new BigDecimal(((Date) this.data).getTime());
376                }
377            } catch (NumberFormatException e) {
378            }
379            return null;
380        }

382        public byte[] getBytes() {
383            if (this.getFieldType().equals(FieldType.BINARY)) {
384                return (byte[]) this.data;
385            } else if (this.getFieldType().equals(FieldType.BLOB)) {
386                try {
387                    return ((Blob) this.data).getBytes(1,
388                            (int) ((Blob) this.data).length());
389                } catch (SQLException e) {
390                    e.getClass();
391                }
392            } else if (this.getFieldType().equals(FieldType.BOOLEAN)) {
393                return ByteBuffer.allocate(1)
394                        .put((byte) (((Boolean) this.data).booleanValue() ? 1 : 0))
395                        .array();
396            } else if (this.getFieldType().equals(FieldType.INTEGER)) {
397                return ByteBuffer.allocate(8).putLong(this.getLong()).array();
398            } else if (this.getFieldType().equals(FieldType.REAL)) {
399                return ByteBuffer.allocate(8).putDouble(this.getDouble()).array();
400            } else if (this.getFieldType().equals(FieldType.STRING)) {
401                return this.data.toString().getBytes();
402            }
403            return null;
404        }

406        public Date getDate() {
407            if (this.getFieldType().equals(FieldType.DATE)) {
408                return (Date) this.data;
409            } else if (this.getFieldType().equals(FieldType.INTEGER)) {
410                return new Date(this.getLong());
411            } else if (this.getFieldType().equals(FieldType.STRING)) {
412                try {
413                    return DateFormat.getDateInstance().parse((String) this.data);
414                } catch (ParseException e) {
415                }
416                try {
417                    return DateFormat.getDateTimeInstance().parse(
418                            (String) this.data);
419                } catch (ParseException e) {
420                }
421                try {
422                    return DateFormat.getTimeInstance().parse((String) this.data);
423                } catch (ParseException e) {
424                }
425                try {
426                    return DateFormat.getInstance().parse((String) this.data);
427                } catch (ParseException e) {
428                }
429            }
430            return null;
431        }

440        public boolean getBool() {
441            if (this.getFieldType().equals(FieldType.BOOLEAN)) {
442                return (Boolean) this.data;
443            } else if (this.getFieldType().equals(FieldType.INTEGER)
444                    || this.getFieldType().equals(FieldType.REAL)
445                    || this.getFieldType().equals(FieldType.DATE)) {
446                return this.getLong() != 0;
447            } else if (this.getFieldType().equals(FieldType.BINARY)
448                    || this.getFieldType().equals(FieldType.BLOB)
449                    || this.getFieldType().equals(FieldType.STRING)) {
450                String s = this.getString();
451                return (s != null) && !s.isEmpty();
452            }
453            return false;
454        }
