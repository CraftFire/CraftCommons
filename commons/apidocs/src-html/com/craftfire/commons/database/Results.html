020    package com.craftfire.commons.database;
021    
022    import java.sql.ResultSet;
023    import java.sql.ResultSetMetaData;
024    import java.sql.SQLException;
025    import java.util.ArrayList;
026    import java.util.List;
027    
028    public class Results {
029        private final String query;
030        private final int rows;
031        private List<DataRow> array;
032    
033        public Results(String query, ResultSet rs) throws SQLException {
034            this.query = query;
035            this.array = new ArrayList<DataRow>();
036    
037            ResultSetMetaData metaData = rs.getMetaData();
038            this.rows = metaData.getColumnCount();
039            while (rs.next()) {
040                DataRow data = new DataRow();
041                for (int i = 1; i <= this.rows; i++) {
042                    data.add(new DataField(i, rs));
043                }
044                this.array.add(data);
045            }
046        }
047    
048        public int getRowsCount() {
049            return this.rows;
050        }
051    
052        public String getQuery() {
053            return this.query;
054        }
055    
056        public List<DataRow> getArray() {
057            return this.array;
058        }
059    
060        public DataRow getFirstResult() {
061            if (this.array.size() > 0) {
062                return this.array.get(0);
063            } else {
064                return null;
065            }
066        }
067    
068        public DataRow getLastResult() {
069            if (this.array.size() > 0) {
070                return this.array.get(this.array.size() - 1);
071            } else {
072                return null;
073            }
074        }
075    }
