020    package com.craftfire.commons.encryption;
021    
022    import java.io.UnsupportedEncodingException;
023    
024    public final class EncryptionUtil {
025        private EncryptionUtil() {
026        }
027    
028        public static String pack(String hex) {
029            StringBuilder buf = new StringBuilder();
030            for (int i = 0; i < hex.length(); i += 2) {
031                char c1 = hex.charAt(i);
032                char c2 = hex.charAt(i + 1);
033                char packed = (char) (hexToInt(c1) * 16 + hexToInt(c2));
034                buf.append(packed);
035            }
036            return buf.toString();
037        }
038    
039        public static String bytesTohex(byte[] data)  {
040            StringBuilder buf = new StringBuilder();
041            for (byte aByte : data) {
042                int halfbyte = (aByte >>> 4) & 0x0F;
043                int twoHalfs = 0;
044                do {
045                    if ((0 <= halfbyte) && (halfbyte <= 9)) {
046                        buf.append((char) ('0' + halfbyte));
047                    } else {
048                        buf.append((char) ('a' + (halfbyte - 10)));
049                    }
050                    halfbyte = aByte & 0x0F;
051                }
052                while (twoHalfs++ < 1);
053            }
054            return buf.toString();
055        }
056    
057        public static int hexToInt(char ch) {
058            if (ch >= '0' && ch <= '9') {
059                return ch - '0';
060            }
061            char nch = Character.toUpperCase(ch);
062            if (nch >= 'A' && nch <= 'F') {
063                return nch - 'A' + 0xA;
064            }
065            throw new IllegalArgumentException("Not a hex character: " + ch);
066        }
067    
068        public static String hexToString(String str) {
069            char[] chars = str.toCharArray();
070            StringBuilder hex = new StringBuilder();
071            for (char aChar : chars) {
072                hex.append(Integer.toHexString(aChar));
073            }
074            return hex.toString();
075        }
076    
077        public static String utf8ToString(byte[] utf8) {
078            try {
079                return new String(utf8, "UTF-8");
080            } catch (UnsupportedEncodingException e) {
081                throw new UnsupportedOperationException("This system doesn't support UTF-8!", e);
082            }
083        }
084    
085        public static byte[] stringToUtf8(String string) {
086            try {
087                return string.getBytes("UTF-8");
088            } catch (UnsupportedEncodingException e) {
089                throw new UnsupportedOperationException("This system doesn't support UTF-8!", e);
090            }
091        }
092    }
