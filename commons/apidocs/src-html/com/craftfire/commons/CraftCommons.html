066        public static String long2ip(Long i) {
067            return ((i >> 24) & 0xFF) + "." +
068                   ((i >> 16) & 0xFF) + "." +
069                   ((i >> 8) & 0xFF) + "." +
070                   (i & 0xFF);
071        }

073        public static Long ip2long(String ip) {
074            String[] ipArray = ip.split("\\.");
075            long num = 0;
076            for (int i = 0; i < ipArray.length; i++) {
077                int power = 3 - i;
078                num += ((Integer.parseInt(ipArray[i]) % 256 * Math.pow(256, power)));
079            }
080            return num;
081        }

111        public static String encrypt(Encryption encryption, Object object, String salt, int iterationCount) {
112            try {
113                String string = (String) object;
114                MessageDigest md = null;
115                String newSalt = salt;
116                if (encryption.equals(Encryption.MD5)) {
117                    md = MessageDigest.getInstance("MD5");
118                } else if (encryption.equals(Encryption.SHA1)) {
119                    md = MessageDigest.getInstance("SHA-1");
120                } else if (encryption.equals(Encryption.SHA256)) {
121                    md = MessageDigest.getInstance("SHA-256");
122                } else if (encryption.equals(Encryption.SHA512)) {
123                    md = MessageDigest.getInstance("SHA-512");
124                } else if (encryption.equals(Encryption.CRC32)) {
125                    Checksum checksum = new CRC32();
126                    checksum.update(string.getBytes(), 0, string.getBytes().length);
127                    return Long.toString(checksum.getValue());
128                } else if (encryption.equals(Encryption.WHIRLPOOL)) {
129                    Whirlpool w = new Whirlpool();
130                    byte[] digest = new byte[Whirlpool.DIGESTBYTES];
131                    w.NESSIEinit();
132                    w.NESSIEadd(string);
133                    w.NESSIEfinalize(digest);
134                    return Whirlpool.display(digest);
135                } else if (encryption.equals(Encryption.PHPASS)) {
136                    PHPass phpass;
137                    if (iterationCount == 0) {
138                        phpass = new PHPass(8);
139                    } else {
140                        phpass = new PHPass(iterationCount);
141                    }
142                    if (newSalt == null || newSalt.isEmpty()) {
143                        newSalt = phpass.gensalt();
144                    }
145                    String hash = phpass.crypt(string, newSalt);
146                    if (hash.length() == 34) {
147                        return hash;
148                    }
149                    return "*";
150                } else if (encryption.equals(Encryption.BLOWFISH)) {
151                    if (newSalt == null || newSalt.isEmpty()) {
152                        if (iterationCount == 0) {
153                            newSalt = BCrypt.gensalt(8);
154                        } else {
155                            newSalt = BCrypt.gensalt(iterationCount);
156                        }
157                    }
158                    return BCrypt.hashpw(string, newSalt);
159                }
160                if (md != null) {
161                    md.update(string.getBytes("ISO-8859-1"), 0, string.length());
162                    byte[] hash = md.digest();
163                    return EncryptionUtil.bytesTohex(hash);
164                } else {
165                    return null;
166                }
167            } catch (GeneralSecurityException e) {
168                throw new IllegalStateException(e);
169            } catch (UnsupportedEncodingException e) {
170                throw new UnsupportedOperationException(e);
171            }
172        }

174        public static Encryption unixHashIdentify(String hash){
175            if (hash.startsWith("$1$") || hash.startsWith("$md5$")) {
176                return Encryption.MD5;
177            } else if (hash.startsWith("$2")) {
178                return Encryption.BLOWFISH;
179            } else if (hash.startsWith("$5$")) {
180                return Encryption.SHA256;
181            } else if (hash.startsWith("$6$")) {
182                return Encryption.SHA512;
183            } else if (hash.startsWith("$P$") || hash.startsWith("$H$")) {
184                return Encryption.PHPASS;
185            }
186            return null;
187        }

226        public static String convertHexToString(String hex) {
227            StringBuilder sb = new StringBuilder();
228            for ( int i=0; i<hex.length()-1; i+=2 ) {
229                String output = hex.substring(i, (i + 2));
230                int decimal = Integer.parseInt(output, 16);
231                sb.append((char)decimal);
232            }
233            return sb.toString();
234        }

236        public static String convertStringToHex(String str) {
237            char[] chars = str.toCharArray();
238            StringBuffer hex = new StringBuffer();
239            for (int i = 0; i < chars.length; i++){
240                hex.append(Integer.toHexString(chars[i]));
241            }
242            return hex.toString();
243        }

338        public static String removeChar(String s, char c) {
339            StringBuilder r = new StringBuilder(s.length());
340            r.setLength(s.length());
341            int current = 0;
342            for (int i = 0; i < s.length(); i++) {
343                char cur = s.charAt(i);
344                if (cur != c) {
345                    r.setCharAt(current++, cur);
346                }
347            }
348            return r.toString();
349        }
