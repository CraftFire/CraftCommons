020    package com.craftfire.commons;
021    
022    import static org.junit.Assert.assertEquals;
023    import static org.junit.Assert.assertFalse;
024    import static org.junit.Assert.assertTrue;
025    
026    import javax.mail.MessagingException;
027    import javax.mail.Session;
028    import javax.mail.internet.AddressException;
029    import javax.mail.internet.InternetAddress;
030    import javax.mail.internet.MimeMessage;
031    
032    import org.junit.Before;
033    import org.junit.Test;
034    
035    import com.craftfire.commons.util.MailSender;
036    
037    public class TestMailSender {
038        MailSender sender;
039    
040        @Before
041        public void init() throws AddressException {
042            this.sender = new MailSender("dev@craftfire.com", "localhost", 25);
043        }
044    
045        @Test
046        public void testSettings() throws AddressException {
047            assertEquals("dev@craftfire.com", this.sender.getSender());
048            assertEquals(new InternetAddress("dev@craftfire.com"), this.sender.getSenderAddress());
049            this.sender.setSender("test@craftfire.com");
050            assertEquals("test@craftfire.com", this.sender.getSender());
051            assertEquals(new InternetAddress("test@craftfire.com"), this.sender.getSenderAddress());
052            this.sender.setSenderAddress(new InternetAddress("bob@craftfire.com"));
053            assertEquals("bob@craftfire.com", this.sender.getSender());
054            assertEquals(new InternetAddress("bob@craftfire.com"), this.sender.getSenderAddress());
055    
056            assertEquals("localhost", this.sender.getHost());
057            this.sender.setHost("smtp.craftfire.com");
058            assertEquals("smtp.craftfire.com", this.sender.getHost());
059    
060            assertEquals(25, this.sender.getPort());
061            this.sender.setPort(225);
062            assertEquals(225, this.sender.getPort());
063    
064            this.sender.setUsername("bob");
065            assertEquals("bob", this.sender.getUsername());
066    
067            this.sender.setPassword("shh...");
068            assertEquals("shh...", this.sender.getPassword());
069    
070            this.sender.setAuth(true);
071            assertTrue(this.sender.isAuth());
072            this.sender.setAuth(false);
073            assertFalse(this.sender.isAuth());
074    
075            this.sender.setSSL(true);
076            assertTrue(this.sender.isSSL());
077            this.sender.setSSL(false);
078            assertFalse(this.sender.isSSL());
079        }
080    
081        @Test
082        public void testSession() {
083            Session s = this.sender.getSession();
084            assertEquals("25", s.getProperty("mail.smtp.port"));
085            assertEquals("localhost", s.getProperty("mail.smtp.host"));
086            assertEquals(String.valueOf(this.sender.isAuth()), s.getProperty("mail.smtp.auth"));
087            this.sender.setSSL(true);
088            this.sender.setHost("smtp.gmail.com");
089            this.sender.setPort(465);
090            this.sender.setUsername("bob");
091            this.sender.setPassword("boo");
092            this.sender.setAuth(true);
093            s = this.sender.getSession();
094            assertEquals("465", s.getProperty("mail.smtp.port"));
095            assertEquals("smtp.gmail.com", s.getProperty("mail.smtps.host"));
096            assertEquals(String.valueOf(this.sender.isAuth()), s.getProperty("mail.smtps.auth"));
097            assertEquals("javax.net.ssl.SSLSocketFactory", s.getProperty("mail.smtp.socketFactory.class"));
098            assertEquals("465", s.getProperty("mail.smtp.socketFactory.port"));
099            assertEquals("false", s.getProperty("mail.smtp.socketFactory.fallback"));
100        }
101    
102        @Test
103        public void testCreateMessage() throws MessagingException {
104            MimeMessage msg = this.sender.createMessage();
105            assertEquals(this.sender.getSenderAddress(), msg.getFrom()[0]);
106        }
107    }
