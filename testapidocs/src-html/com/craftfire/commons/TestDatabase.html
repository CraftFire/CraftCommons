123        @Test
124        public void testInsert() throws SQLException {
125            int prevId = datamanager.getLastID("id", wrtable);
126            Date now = new Date();
127            Map<String, Object> data = new HashMap<String, Object>();
128            data.put("TXT", "commons" + randomInt);
129            data.put("x", randomInt + 1);
130            data.put("d", now);
131            data.put("b", null);
132            datamanager.insertFields(data, wrtable);
133            int id = datamanager.getLastID("id", wrtable);
134            assertTrue(id > prevId);
135            assertEquals(randomInt + 1, datamanager.getIntegerField(wrtable, "x", "`id` = '" + id + "'"));
136            assertEquals("commons" + randomInt, datamanager.getStringField(wrtable, "txt", "`id` = '" + id + "'"));
137            assertEquals(now.getTime(), datamanager.getDateField(wrtable, "d", "`id` = '" + id + "'").getTime());
138            assertTrue(datamanager.getField(ValueType.UNKNOWN, wrtable, "b", "`id` = '" + id + "'").isNull());
139        }

141        @Test
142        public void testUpdate() throws SQLException {
143            String oldString = datamanager.getStringField(wrtable, "txt", "`id` = '1'");
144            String testString = "crafttest" + (randomInt + 2);
145            Date oldDate = datamanager.getDateField(wrtable, "d", "`id` = '1'");
146            Date testDate = new Date(1356120741);
147            int oldX = datamanager.getIntegerField(wrtable, "x", "`id` = '1'");
148            Map<String, Object> data = new HashMap<String, Object>();
149            data.put("txt", testString);
150            data.put("d", testDate);
151            data.put("x", null);
152            datamanager.updateFields(data, wrtable, "`id` = '1'");
153            assertEquals(testString, datamanager.getStringField(wrtable, "txt", "`id` = '1'"));
154            assertTrue(datamanager.getField(ValueType.UNKNOWN, wrtable, "x", "`id` = '1'").isNull());
155            assertEquals(testDate, datamanager.getDateField(wrtable, "d", "`id` = '1'"));
156            datamanager.updateField(wrtable, "txt", oldString, "`id` = '1'");
157            datamanager.updateField(wrtable, "d", oldDate, "`id` = '1'");
158            datamanager.executeQueryVoid("UPDATE `" + wrtable + "` SET `x` = '" + oldX + "' WHERE `id` = '1'");
159            assertEquals(oldString, datamanager.getStringField(wrtable, "txt", "`id` = '1'"));
160            assertEquals(oldDate, datamanager.getDateField(wrtable, "d", "`id` = '1'"));
161        }

232        public void testBInt() throws SQLException, ParseException, IOException {
233            final String name = "bint";
234            final BigInteger expected = new BigInteger("487250340273948");
235            final Date expectedDate = new Date(487250340273948L);
236            final byte[] expectedBytes = { 0, 1, -69, 38, -49, 114, -33, 28 };
237            DataRow row = datamanager.getResults("SELECT `" + name + "` FROM `" + table + "` LIMIT 1").getFirstResult();
238    
239            // DataRow.getField()
240            DataField field = row.get(name);
241            assertEquals(expected, field.getBigInt());
242            assertTrue(field.getBool());
243            assertArrayEquals(expectedBytes, field.getBytes());
244            assertEquals(expectedDate, field.getDate());
245            assertEquals(new BigDecimal(expected), field.getDecimal());
246            assertEquals(expected.doubleValue(), field.getDouble(), 0);
247            assertEquals(expected.floatValue(), field.getFloat(), 0);
248            assertEquals(expected.intValue(), field.getInt());
249            assertEquals(expected.longValue(), field.getLong());
250            assertEquals("487250340273948", field.getString());
251    
252            // DataManager.getField()
253            field = datamanager.getField(ValueType.UNKNOWN, table, name, "1");
254            assertEquals(expected, field.getBigInt());
255            assertTrue(field.getBool());
256            assertArrayEquals(expectedBytes, field.getBytes());
257            assertEquals(expectedDate, field.getDate());
258            assertEquals(new BigDecimal(expected), field.getDecimal());
259            assertEquals(expected.doubleValue(), field.getDouble(), 0);
260            assertEquals(expected.floatValue(), field.getFloat(), 0);
261            assertEquals(expected.intValue(), field.getInt());
262            assertEquals(expected.longValue(), field.getLong());
263            assertEquals("487250340273948", field.getString());
264    
265            // DataManager.get<Kinda>Field()
266            assertTrue(datamanager.getBooleanField(table, name, "1"));
267            assertEquals(new String(expectedBytes), datamanager.getBinaryField(table, name, "1"));
268            // assertNotNull(datamanager.getBlobField(table, name, "1"));
269            InputStream stream = datamanager.getBlobField(table, name, "1").getBinaryStream();
270            assertEquals(new String(expectedBytes), CraftCommons.convertStreamToString(stream));
271            assertEquals(expected.doubleValue(), datamanager.getDoubleField(table, name, "1"), 0);
272            assertEquals("487250340273948", datamanager.getStringField(table, name, "1"));
273    
274            // DataRow.get<Kinda>Field()
275            assertEquals(expected, row.getBigIntField(name));
276            assertTrue(row.getBoolField(name));
277            assertArrayEquals(expectedBytes, row.getBinaryField(name));
278            assertEquals(expectedDate, row.getDateField(name));
279            assertEquals(new BigDecimal(expected), row.getDecimalField(name));
280            assertEquals(expected.doubleValue(), row.getDoubleField(name), 0);
281            assertEquals(expected.floatValue(), row.getFloatField(name), 0);
282            assertEquals(expected.intValue(), row.getIntField(name));
283            assertEquals(expected.longValue(), row.getLongField(name), 0);
284            assertEquals("487250340273948", row.getStringField(name));
