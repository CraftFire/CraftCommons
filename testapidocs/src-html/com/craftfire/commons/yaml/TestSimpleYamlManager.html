519        class IsReaderThatContains extends ArgumentMatcher<Reader> {
520            private final String string;
521            private List<Reader> readers = new ArrayList<Reader>();
522    
523            public IsReaderThatContains(String string) {
524                this.string = string;
525            }
526    
527            @Override
528            public boolean matches(Object argument) {
529                if (argument instanceof Reader) {
530                    if (this.readers.contains(argument)) {
531                        return true;
532                    }
533                    char[] buf = new char[this.string.length()];
534                    try {
535                        ((Reader) argument).read(buf);
536                        if (new String(buf).equals(this.string)) {
537                            this.readers.add((Reader) argument);
538                            return true;
539                        }
540                    } catch (IOException e) {
541                        e.printStackTrace();
542                    }
543                }
544                return false;
545            }
546        }

548        class IsStreamThatContains extends ArgumentMatcher<InputStream> {
549            private final byte[] bytes;
550            private List<InputStream> streams = new ArrayList<InputStream>();
551    
552            public IsStreamThatContains(byte[] bytes) {
553                this.bytes = bytes;
554            }
555    
556            @Override
557            public boolean matches(Object argument) {
558                if (argument instanceof InputStream) {
559                    if (this.streams.contains(argument)) {
560                        return true;
561                    }
562                    byte[] buf = new byte[this.bytes.length];
563                    try {
564                        ((InputStream) argument).read(buf);
565                        if (Arrays.equals(buf, this.bytes)) {
566                            this.streams.add((InputStream) argument);
567                            return true;
568                        }
569                    } catch (IOException e) {
570                        e.printStackTrace();
571                    }
572                }
573                return false;
574            }
575        }
