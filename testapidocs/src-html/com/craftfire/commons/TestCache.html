020    package com.craftfire.commons;
021    
022    import static org.junit.Assert.assertEquals;
023    import static org.junit.Assert.assertNull;
024    
025    import java.util.Date;
026    
027    import org.junit.Test;
028    
029    import com.craftfire.commons.cache.CacheManager;
030    import com.craftfire.commons.cache.MetadatableCacheItem;
031    
032    public class TestCache {
033    
034        @Test
035        public void testPutGet() {
036            CacheManager mgr = new CacheManager();
037            mgr.setEnabled(true);
038            Date now = new Date();
039            int id = mgr.put(now);
040            assertEquals(now, mgr.get(id));
041        }
042    
043        @Test
044        public void testMetaPutGet() {
045            CacheManager mgr = new CacheManager();
046            mgr.setEnabled(true);
047            Date now = new Date();
048            int id = mgr.putMetadatable(now);
049            ((MetadatableCacheItem) mgr.getItem(id)).setMetaData("test.meta", "This is the freaky metadata");
050            assertEquals(now, mgr.get(id));
051            assertEquals("This is the freaky metadata", ((MetadatableCacheItem) mgr.getItem(id)).getMetaData("test.meta"));
052            assertNull(((MetadatableCacheItem) mgr.getItem(id)).getMetaData("test_meta"));
053            ((MetadatableCacheItem) mgr.getItem(id)).removeMetaData("test.meta");
054            assertNull(((MetadatableCacheItem) mgr.getItem(id)).getMetaData("test.meta"));
055        }
056    
057        @Test
058        public void testGroupPutGet() {
059            CacheManager mgr = new CacheManager();
060            mgr.setEnabled(true);
061            Date now = new Date();
062            int id = mgr.put("My fancy Group", now);
063            assertEquals(now, mgr.get("My fancy Group", id));
064        }
065    
066        @Test
067        public void testGroupMetaPutGet() {
068            CacheManager mgr = new CacheManager();
069            mgr.setEnabled(true);
070            Date now = new Date();
071            int id = mgr.putMetadatable("GrOuP", now);
072            mgr.getMetadatableItem("GrOuP", id).setMetaData("test.meta", "This is the freaky metadata");
073            assertEquals(now, mgr.get("GrOuP", id));
074            assertEquals("This is the freaky metadata", mgr.getMetadatableItem("GrOuP", id).getMetaData("test.meta"));
075            assertNull(mgr.getMetadatableItem("GrOuP", id).getMetaData("test_meta"));
076            mgr.getMetadatableItem("GrOuP", id).removeMetaData("test.meta");
077            assertNull(mgr.getMetadatableItem("GrOuP", id).getMetaData("test.meta"));
078        }
079    
080    }
