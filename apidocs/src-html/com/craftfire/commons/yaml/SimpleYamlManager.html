265            if (exist(node)) {
266                try {
267                    boolean value = this.root.getNode(node).getBool(defaultValue);
268                    getLogger().debug("Found node '" + node + "' with Boolean value '" + value + "', default value is '" + defaultValue + "'.");
269                    return value;
270                } catch (YamlException e) {
271                    getLogger().stackTrace(e);
272                }
273            }
274            getLogger().debug("Could not find node '" + node + "', returning default value instead '" + defaultValue + "'.");
275            return defaultValue;
276        }

421        /* (non-Javadoc)
422         * @see com.craftfire.commons.yaml.YamlManager#save()
423         */
424        @Override
425        public boolean save() {
426            if (this.file == null) {
427                return false;
428            }
429            try {
430                this.yaml.dump(this.root.dump(), new FileWriter(this.file));
431            } catch (IOException e) {
432                getLogger().stackTrace(e);
433                return false;
434            }
435            return true;
436        }

438        /* (non-Javadoc)
439         * @see com.craftfire.commons.yaml.YamlManager#reload()
440         */
441        @Override
442        public boolean load() {
443            try {
444                if (this.reader != null) {
445                    load(this.reader);
446                    this.reader = null;
447                    return true;
448                }
449                if (this.resource != null) {
450                    load(this.resource);
451                    return true;
452                }
453                if (this.file != null) {
454                    load(this.file);
455                    return true;
456                }
457            } catch (IOException e) {
458                getLogger().stackTrace(e);
459            }
460            return false;
461        }

463        /**
464         * Loads yaml from the given file and sets is as the root node.
465         * <p>
466         * Tries to create the file if doesn't exist.
467         * 
468         * @param  file        file to load the yaml from
469         * @throws IOException if an IOException occurred
470         */
471        protected void load(File file) throws IOException {
472            if (file == null) {
473                getLogger().error("File '" + file + "' is null, nodes could not be loaded from the yaml file.");
474                return;
475            } else if (!file.exists()) {
476                try {
477                    if (file.getParentFile() != null) {
478                        file.getParentFile().mkdirs();
479                    }
480                    file.createNewFile();
481                } catch (IOException e) {
482                    getLogger().error("Unable to create file '" + file);
483                    getLogger().stackTrace(e);
484                    return;
485                }
486            }
487            getLogger().debug("Loading nodes from file '" + file.getAbsoluteFile() + "'.");
488            load(new FileInputStream(file));
489        }
