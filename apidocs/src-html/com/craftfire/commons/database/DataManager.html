210        protected boolean setURL() {
211            switch (this.datatype) {
212                case MYSQL:
213                    if (this.host == null || this.database == null) {
214                        getLogger().debug("Could not set mySQL URL. Host: " + this.host + ", Database: " + this.database);
215                        return false;
216                    }
217                    this.url = "jdbc:mysql://" + this.host + "/" + this.database
218                            + "?zeroDateTimeBehavior=convertToNull"
219                            + "&jdbcCompliantTruncation=false"
220                            + "&autoReconnect=true"
221                            + "&characterEncoding=UTF-8"
222                            + "&characterSetResults=UTF-8";
223                    outputDrivers();
224                    return true;
225                case H2:
226                    if (this.directory == null || this.database == null) {
227                        getLogger().debug("Could not set H2 URL. Host: " + this.directory + ", Database: " + this.database);
228                        return false;
229                    }
230                    this.url = "jdbc:h2:" + this.directory + this.database + ";AUTO_RECONNECT=TRUE";
231                    outputDrivers();
232                    return true;
233            }
234            return false;
235        }

505        public DataField getField(ValueType type, String query) throws SQLException {
506            try {
507                this.rs = getResultSet(query);
508                if (this.rs.next()) {
509                    Object value;
510                    if (type.equals(ValueType.STRING)) {
511                        value = this.rs.getString(1);
512                    } else if (type.equals(ValueType.INTEGER)) {
513                        value = this.rs.getInt(1);
514                    } else if (type.equals(ValueType.DATE)) {
515                        value = this.rs.getTimestamp(1);
516                    } else if (type.equals(ValueType.BLOB)) {
517                        value = this.rs.getBlob(1);
518                    } else if (type.equals(ValueType.BINARY)) {
519                        value = this.rs.getBytes(1);
520                    } else if (type.equals(ValueType.BOOLEAN)) {
521                        value = this.rs.getBoolean(1);
522                    } else if (type.equals(ValueType.REAL)) {
523                        value = this.rs.getDouble(1);
524                    } else if (type.equals(ValueType.UNKNOWN)) {
525                        DataField dataField = new DataField(1, this.rs);
526                        close();
527                        return dataField;
528                    } else {
529                        close();
530                        return null;
531                    }
532                    DataField dataField = new DataField(type, this.rs.getMetaData().getColumnDisplaySize(1), value);
533                    close();
534                    return dataField;
535                }
536            } finally {
537                close();
538            }
539            return null;
540        }

543            connect();
544            log(query);

561        public void updateBlob(String table, String field, String where, String data) {
562            try {
563                String query = "UPDATE `" + getPrefix() + table + "` " + "SET `" + field + "` = ? " + "WHERE " + where;
564                byte[] array = data.getBytes();
565                ByteArrayInputStream inputStream = new ByteArrayInputStream(array);
566                connect();
567                log(query);
568                if (this.con == null) {
569                    throw new SQLException("Unable to connect to the database");
570                }
571                this.stmt = this.con.createStatement();
572                this.pStmt = this.con.prepareStatement(query);
573                this.pStmt.setBlob(1, inputStream, array.length);
574                this.pStmt.executeUpdate();
575                close();
576            } catch (SQLException e) {
577                getLogger().stackTrace(e);
578            }
579        }

600        public TableModel resultSetToTableModel(String query) {
601            try {
602                this.rs = getResultSet(query);
603                ResultSetMetaData metaData = this.rs.getMetaData();
604                int numberOfColumns = metaData.getColumnCount();
605                Vector<String> columnNames = new Vector<String>();
606                for (int column = 0; column < numberOfColumns; column++) {
607                    columnNames.addElement(metaData.getColumnLabel(column + 1));
608                }
609                Vector<Vector<Object>> rows = new Vector<Vector<Object>>();
610                while (this.rs.next()) {
611                    Vector<Object> newRow = new Vector<Object>();
612                    for (int i = 1; i <= numberOfColumns; i++) {
613                        newRow.addElement(this.rs.getObject(i));
614                    }
615                    rows.addElement(newRow);
616                }
617                close();
618                return new DefaultTableModel(rows, columnNames);
619            } catch (Exception e) {
620                getLogger().stackTrace(e);
621                close();
622                return null;
623            }
624        }

699        private void outputDrivers() {
700            if (getLogger().isDebug()) {
701                getLogger().debug("Checking DriverManager drivers.");
702                Enumeration<Driver> driverList = DriverManager.getDrivers();
703                int count = 0;
704                while (driverList.hasMoreElements()) {
705                    Driver driverClass = driverList.nextElement();
706                    getLogger().debug("Found driver #" + (count + 1) + ": " + driverClass.getClass().getName());
707                    count++;
708                }
709                getLogger().debug("Found " + count + " drivers in DriverManager.");
710            }
711        }

739        public void connect() {
740            if (this.url == null && !setURL()) {
741                return;
742            }
743            if (this.con != null && isConnected()) {
744                return;
745            }
746            long start = System.currentTimeMillis();
747            try {
748                switch (this.datatype) {
749                    case MYSQL:
750                        getLogger().debug("Connecting to MYSQL with URL '" + this.url + "'.");
751                        Class.forName("com.mysql.jdbc.Driver");
752                        this.con = DriverManager.getConnection(this.url, this.username, this.password);
753                        break;
754                    case H2:
755                        getLogger().debug("Connecting to H2 with URL '" + this.url + "'.");
756                        Class.forName("org.h2.Driver");
757                        this.con = DriverManager.getConnection(this.url, this.username, this.password);
758                        break;
759                }
760                this.startup = System.currentTimeMillis() / 1000;
761            } catch (ClassNotFoundException e) {
762                getLogger().error("Could not connect to the database due to no driver could be found for '" + this.datatype + "'.");
763                getLogger().debug("Connection attempt took " + new TimeUtil((System.currentTimeMillis() - start) / 1000).toString() + ".");
764                getLogger().stackTrace(e);
765                return;
766            } catch (SQLException e) {
767                getLogger().error("Could not connect to the database for '" + this.datatype + "' due to a SQL Exception.");
768                getLogger().debug("Connection attempt took " + new TimeUtil((System.currentTimeMillis() - start) / 1000).toString() + ".");
769                getLogger().stackTrace(e);
770                return;
771            }
772            getLogger().debug("Took " + new TimeUtil((System.currentTimeMillis() - start) / 1000).toString() +
773                    " to establish a connection for '" + this.datatype + "'.");
774        }

780        public void close(boolean force) {
781            try {
782                if (this.rs != null) {
783                    this.rs.close();
784                    this.rs = null;
785                }
786                if (this.pStmt != null) {
787                    this.pStmt.close();
788                    this.pStmt = null;
789                }
790                if (this.stmt != null) {
791                    this.stmt.close();
792                    this.stmt = null;
793                }
794            } catch (SQLException e) {
795                getLogger().stackTrace(e);
796            }
797            if (this.keepAlive && !this.reconnect && !force) {
798                if (this.timeout == 0) {
799                    return;
800                } else if ((System.currentTimeMillis() / 1000) < (this.startup + this.timeout)) {
801                    return;
802                }
803            }
804            long start = System.currentTimeMillis();
805            try {
806                getLogger().debug("Closing connection for '" + this.datatype + "'. Uptime: " + new TimeUtil(getUptime()).toString() + ". Queries: " + getQueriesCount() + ".");
807                if (this.con != null) {
808                    this.con.close();
809                    this.con = null;
810                }
811                if (this.keepAlive && !force) {
812                    connect();
813                }
814            } catch (SQLException e) {
815                getLogger().error("Could not close the connection to the database for '" + this.datatype + "' due to a SQL Exception.");
816                getLogger().debug("Closing connection attempt took " + new TimeUtil((System.currentTimeMillis() - start) / 1000).toString() + ".");
817                getLogger().stackTrace(e);
818                return;
819            }
820            getLogger().debug("Took " + new TimeUtil((System.currentTimeMillis() - start) / 1000).toString() +
821                    " to CLOSE connection for '" + this.datatype + "'.");
822        }

824        public void reconnect() {
825            getLogger().debug("Attempting to reconnect connection for '" + this.datatype + "'.");
826            long start = System.currentTimeMillis();
827            this.reconnect = true;
828            close();
829            connect();
830            this.reconnect = false;
831            getLogger().debug("Took " + new TimeUtil((System.currentTimeMillis() - start) / 1000).toString() +
832                    " to attempt a reconnection connection for '" + this.datatype + "'.");
833        }

835        protected String updateFieldsString(Map<String, Object> data) {
836            String query = " SET", suffix = ",";
837            int i = 1;
838            Iterator<Entry<String, Object>> it = data.entrySet().iterator();
839            while (it.hasNext()) {
840                Map.Entry<String, Object> pairs = it.next();
841                if (i == data.size()) {
842                    suffix = "";
843                }
844                query += " `" + pairs.getKey() + "` =  " + fieldValueToString(pairs.getValue()) + suffix;
845                i++;
846            }
847            return query;
848        }

850        protected String insertFieldString(Map<String, Object> data) {
851            String fields = "", values = "", query = "", suffix = ",";
852            int i = 1;
853            Iterator<Entry<String, Object>> it = data.entrySet().iterator();
854            while (it.hasNext()) {
855                Map.Entry<String, Object> pairs = it.next();
856                if (i == data.size()) {
857                    suffix = "";
858                }
859                fields += " `" + pairs.getKey() + "`" + suffix;
860                values += fieldValueToString(pairs.getValue()) + suffix;
861                i++;
862            }
863            query = "(" + fields + ") VALUES (" + values + ")";
864            return query;
865        }

867        protected String fieldValueToString(Object value) {
868            Object val = value;
869            if (val instanceof Date) {
870                val = new Timestamp(((Date) val).getTime());
871            }
872            if (val == null) {
873                return "NULL";
874            } else {
875                return "'" + val.toString().replaceAll("'", "''") + "'";
876            }
877        }
