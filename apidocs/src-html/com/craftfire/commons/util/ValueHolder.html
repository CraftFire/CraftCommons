020    package com.craftfire.commons.util;
021    
022    import java.math.BigDecimal;
023    import java.math.BigInteger;
024    import java.sql.Blob;
025    import java.util.Date;
026    
027    /**
028     * An interface for everything that holds different types of values and (optionally) has a name.
029     */
030    public interface ValueHolder {
031    
032        /**
033         * Returns the name of the ValueHolder
034         * 
035         * @return the name
036         */
037        String getName();
038    
039        /**
040         * Returns the type of the value.
041         * 
042         * @return type of the value
043         */
044        ValueType getType();
045    
046        /**
047         * Returns the value.
048         * 
049         * @return the value
050         */
051        Object getValue();
052    
053        /**
054         * Returns the value as {@code String}.
055         * 
056         * @return the value as {@code String}
057         */
058        String getString();
059    
060        /**
061         * Returns the value as {@code String}.
062         * 
063         * @param defaultValue  the default value to return if the value is not a {@code String} and can't be converted to it
064         * @return              the value as {@code String} or {@code defaultValue} if the value is not a {@code String} and can't be converted
065         */
066        String getString(String defaultValue);
067    
068        /**
069         * Returns the value as {@code int}.
070         * 
071         * @return the value as {@code int}
072         */
073        int getInt();
074    
075        /**
076         * Returns the value as {@code int}.
077         * 
078         * @param defaultValue  the default value to return if the value is not an {@code int} and can't be converted to it
079         * @return              the value as {@code int} or {@code defaultValue} if the value is not an {@code int} and can't be converted
080         */
081        int getInt(int defaultValue);
082    
083        /**
084         * Returns the value as {@code long}.
085         * 
086         * @return the value as {@code long}
087         */
088        long getLong();
089    
090        /**
091         * Returns the value as {@code long}.
092         * 
093         * @param defaultValue  the default value to return if the value is not a {@code long} and can't be converted to it
094         * @return              the value as {@code long} or {@code defaultValue} if the value is not a {@code long} and can't be converted
095         */
096        long getLong(long defaultValue);
097    
098        /**
099         * Returns the value as {@code BigInteger}.
100         * 
101         * @return the value as {@code BigInteger}
102         */
103        BigInteger getBigInt();
104    
105        /**
106         * Returns the value as {@code BigInteger}.
107         * 
108         * @param defaultValue  the default value to return if the value is not a {@code BigInteger} and can't be converted to it
109         * @return              the value as {@code BigInteger} or {@code defaultValue} if the value is not a {@code BigInteger} and can't be converted
110         */
111        BigInteger getBigInt(BigInteger defaultValue);
112    
113        /**
114         * Returns the value as {@code double}.
115         * 
116         * @return the value as {@code double}
117         */
118        double getDouble();
119    
120        /**
121         * Returns the value as {@code double}.
122         * 
123         * @param defaultValue  the default value to return if the value is not a {@code double} and can't be converted to it
124         * @return              the value as {@code double} or {@code defaultValue} if the value is not a {@code double} and can't be converted
125         */
126        double getDouble(double defaultValue);
127    
128        /**
129         * Returns the value as {@code float}.
130         * 
131         * @return the value as {@code float}
132         */
133        float getFloat();
134    
135        /**
136         * Returns the value as {@code float}.
137         * 
138         * @param defaultValue  the default value to return if the value is not a {@code float} and can't be converted to it
139         * @return              the value as {@code float} or {@code defaultValue} if the value is not a {@code float} and can't be converted
140         */
141        float getFloat(float defaultValue);
142    
143        /**
144         * Returns the value as {@code BigDecimal}.
145         * 
146         * @return the value as {@code BigDecimal}
147         */
148        BigDecimal getDecimal();
149    
150        /**
151         * Returns the value as {@code BigDecimal}.
152         * 
153         * @param defaultValue  the default value to return if the value is not a {@code BigDecimal} and can't be converted to it
154         * @return              the value as {@code BigDeciaml} or {@code defaultValue} if the value is not a {@code BigDecimal} and can't be converted
155         */
156        BigDecimal getDecimal(BigDecimal defaultValue);
157    
158        /**
159         * Returns the value as {@code byte} array.
160         * 
161         * @return the value as {@code byte} array
162         */
163        byte[] getBytes();
164    
165        /**
166         * Returns the value as {@code byte} array.
167         * 
168         * @param defaultValue  the default value to return if the value is not a {@code byte} array and can't be converted to it
169         * @return              the value as {@code byte} array or {@code defaultValue} if the value is not a {@code byte} array and can't be converted
170         */
171        byte[] getBytes(byte[] defaultValue);
172    
173        /**
174         * Returns the value as {@code Date}.
175         * 
176         * @return the value as {@code Date}
177         */
178        Date getDate();
179    
180        /**
181         * Returns the value as {@code Date}.
182         * 
183         * @param defaultValue  the default value to return if the value is not a {@code Date} and can't be converted to it
184         * @return              the value as {@code Date} or {@code defaultValue} if the value is not a {@code Date} and can't be converted
185         */
186        Date getDate(Date defaultValue);
187    
188        /**
189         * Returns the value as {@code Blob}.
190         * 
191         * @return the value as {@code Blob}
192         */
193        Blob getBlob();
194    
195        /**
196         * Returns the value as {@code Blob}.
197         * 
198         * @param defaultValue  the default value to return if the value is not a {@code Blob} and can't be converted to it
199         * @return              the value as {@code Blob} or {@code defaultValue} if the value is not a {@code Blob} and can't be converted
200         */
201        Blob getBlob(Blob defaultValue);
202    
203        /**
204         * Returns the value as {@code boolean}.
205         * 
206         * @return the value as {@code boolean}
207         */
208        boolean getBool();
209    
210        /**
211         * Returns the value as {@code boolean}.
212         * 
213         * @param defaultValue  the default value to return if the value is not a {@code boolean} and can't be converted to it
214         * @return              the value as {@code boolean} or {@code defaultValue} if the value is not a {@code boolean} and can't be converted
215         */
216        boolean getBool(boolean defaultValue);
217    
218        /**
219         * Checks if the value is null.
220         * 
221         * @return {@code true} if it's {@code null}, {@code false} otherwise
222         */
223        boolean isNull();
224    
225        /**
226         * Checks if the value is unsigned.
227         * 
228         * @return {@code true} if it's unsigned, {@code false} otherwise
229         */
230        boolean isUnsigned();
231    
232    }
