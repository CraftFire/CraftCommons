040    /**
041    * BCrypt implements OpenBSD-style Blowfish password hashing using
042    * the scheme described in "A Future-Adaptable Password Scheme" by
043    * Niels Provos and David Mazieres.
044    * <p>
045    * This password hashing system tries to thwart off-line password
046    * cracking using a computationally-intensive hashing algorithm,
047    * based on Bruce Schneier's Blowfish cipher. The work factor of
048    * the algorithm is parameterised, so it can be increased as
049    * computers get faster.
050    * <p>
051    * Usage is really simple. To hash a password for the first time,
052    * call the hashpw method with a random salt, like this:
053    * <p>
054    * <code>
055    * String pw_hash = BCrypt.hashpw(plain_password, BCrypt.gensalt()); <br />
056    * </code>
057    * <p>
058    * To check whether a plaintext password matches one that has been
059    * hashed previously, use the checkpw method:
060    * <p>
061    * <code>
062    * if (BCrypt.checkpw(candidate_password, stored_hash))<br />
063    * &nbsp;&nbsp;&nbsp;&nbsp;System.out.println("It matches");<br />
064    * else<br />
065    * &nbsp;&nbsp;&nbsp;&nbsp;System.out.println("It does not match");<br />
066    * </code>
067    * <p>
068    * The gensalt() method takes an optional parameter (log_rounds)
069    * that determines the computational complexity of the hashing:
070    * <p>
071    * <code>
072    * String strong_salt = BCrypt.gensalt(10)<br />
073    * String stronger_salt = BCrypt.gensalt(12)<br />
074    * </code>
075    * <p>
076    * The amount of work increases exponentially (2**log_rounds), so
077    * each increment is twice as much work. The default log_rounds is
078    * 10, and the valid range is 4 to 31.
079    *
080    * @author Damien Miller
081    * @version 0.2
082    */

395        /**
396         * Encode a byte array using bcrypt's slightly-modified base64
397         * encoding scheme. Note that this is *not* compatible with
398         * the standard MIME-base64 encoding.
399         *
400         * @param d    the byte array to encode
401         * @param len    the number of bytes to encode
402         * @return    base64-encoded string
403         * @exception IllegalArgumentException if the length is invalid
404         */
405        private static String encode_base64(byte d[], int len)
406            throws IllegalArgumentException {
407            int off = 0;
408            StringBuffer rs = new StringBuffer();
409            int c1, c2;
410    
411            if (len <= 0 || len > d.length) {
412                throw new IllegalArgumentException ("Invalid len");
413            }
414    
415            while (off < len) {
416                c1 = d[off++] & 0xff;
417                rs.append(base64_code[(c1 >> 2) & 0x3f]);
418                c1 = (c1 & 0x03) << 4;
419                if (off >= len) {
420                    rs.append(base64_code[c1 & 0x3f]);
421                    break;
422                }
423                c2 = d[off++] & 0xff;
424                c1 |= (c2 >> 4) & 0x0f;
425                rs.append(base64_code[c1 & 0x3f]);
426                c1 = (c2 & 0x0f) << 2;
427                if (off >= len) {
428                    rs.append(base64_code[c1 & 0x3f]);
429                    break;
430                }
431                c2 = d[off++] & 0xff;
432                c1 |= (c2 >> 6) & 0x03;
433                rs.append(base64_code[c1 & 0x3f]);
434                rs.append(base64_code[c2 & 0x3f]);
435            }
436            return rs.toString();
437        }

441         * range-checking againt conversion table
442         * @param x    the base64-encoded value

445        private static byte char64(char x) {
446            if (x < 0 || x > index_64.length) {
447                return -1;

452        /**
453         * Decode a string encoded using bcrypt's base64 scheme to a
454         * byte array. Note that this is *not* compatible with
455         * the standard MIME-base64 encoding.
456         * @param s    the string to decode
457         * @param maxolen    the maximum number of bytes to decode
458         * @return    an array containing the decoded bytes
459         * @throws IllegalArgumentException if maxolen is invalid
460         */
461        private static byte[] decode_base64(String s, int maxolen)
462            throws IllegalArgumentException {
463            StringBuffer rs = new StringBuffer();
464            int off = 0, slen = s.length(), olen = 0;
465            byte ret[];
466            byte c1, c2, c3, c4, o;
467    
468            if (maxolen <= 0) {
469                throw new IllegalArgumentException ("Invalid maxolen");
470            }
471    
472            while (off < slen - 1 && olen < maxolen) {
473                c1 = char64(s.charAt(off++));
474                c2 = char64(s.charAt(off++));
475                if (c1 == -1 || c2 == -1) {
476                    break;
477                }
478                o = (byte)(c1 << 2);
479                o |= (c2 & 0x30) >> 4;
480                rs.append((char)o);
481                if (++olen >= maxolen || off >= slen) {
482                    break;
483                }
484                c3 = char64(s.charAt(off++));
485                if (c3 == -1) {
486                    break;
487                }
488                o = (byte)((c2 & 0x0f) << 4);
489                o |= (c3 & 0x3c) >> 2;
490                rs.append((char)o);
491                if (++olen >= maxolen || off >= slen) {
492                    break;
493                }
494                c4 = char64(s.charAt(off++));
495                o = (byte)((c3 & 0x03) << 6);
496                o |= c4;
497                rs.append((char)o);
498                ++olen;
499            }
500    
501            ret = new byte[olen];
502            for (off = 0; off < olen; off++) {
503                ret[off] = (byte)rs.charAt(off);
504            }
505            return ret;
506        }

508        /**
509         * Blowfish encipher a single 64-bit block encoded as
510         * two 32-bit halves
511         * @param lr    an array containing the two 32-bit half blocks
512         * @param off    the position in the array of the blocks
513         */
514        private final void encipher(int lr[], int off) {
515            int i, n, l = lr[off], r = lr[off + 1];
516    
517            l ^= this.P[0];
518            for (i = 0; i <= BLOWFISH_NUM_ROUNDS - 2;) {
519                // Feistel substitution on left word
520                n = this.S[(l >> 24) & 0xff];
521                n += this.S[0x100 | ((l >> 16) & 0xff)];
522                n ^= this.S[0x200 | ((l >> 8) & 0xff)];
523                n += this.S[0x300 | (l & 0xff)];
524                r ^= n ^ this.P[++i];
525    
526                // Feistel substitution on right word
527                n = this.S[(r >> 24) & 0xff];
528                n += this.S[0x100 | ((r >> 16) & 0xff)];
529                n ^= this.S[0x200 | ((r >> 8) & 0xff)];
530                n += this.S[0x300 | (r & 0xff)];
531                l ^= n ^ this.P[++i];
532            }
533            lr[off] = r ^ this.P[BLOWFISH_NUM_ROUNDS + 1];
534            lr[off + 1] = l;
535        }

537        /**
538         * Cycically extract a word of key material
539         * @param data    the string to extract the data from
540         * @param offp    a "pointer" (as a one-entry array) to the
541         * current offset into data
542         * @return    the next word of material from data
543         */
544        private static int streamtoword(byte data[], int offp[]) {
545            int i;
546            int word = 0;
547            int off = offp[0];
548    
549            for (i = 0; i < 4; i++) {
550                word = (word << 8) | (data[off] & 0xff);
551                off = (off + 1) % data.length;
552            }
553    
554            offp[0] = off;
555            return word;
556        }

569        /**
570         * Key the Blowfish cipher
571         * @param key    an array containing the key
572         */
573        private void key(byte key[]) {
574            int i;
575            int koffp[] = { 0 };
576            int lr[] = { 0, 0 };
577            int plen = this.P.length, slen = this.S.length;
578    
579            for (i = 0; i < plen; i++) {
580                this.P[i] = this.P[i] ^ streamtoword(key, koffp);
581            }
582    
583            for (i = 0; i < plen; i += 2) {
584                encipher(lr, 0);
585                this.P[i] = lr[0];
586                this.P[i + 1] = lr[1];
587            }
588    
589            for (i = 0; i < slen; i += 2) {
590                encipher(lr, 0);
591                this.S[i] = lr[0];
592                this.S[i + 1] = lr[1];
593            }
594        }

596        /**
597         * Perform the "enhanced key schedule" step described by
598         * Provos and Mazieres in "A Future-Adaptable Password Scheme"
599         * http://www.openbsd.org/papers/bcrypt-paper.ps
600         * @param data    salt information
601         * @param key    password information
602         */
603        private void ekskey(byte data[], byte key[]) {
604            int i;
605            int koffp[] = { 0 }, doffp[] = { 0 };
606            int lr[] = { 0, 0 };
607            int plen = this.P.length, slen = this.S.length;
608    
609            for (i = 0; i < plen; i++) {
610                this.P[i] = this.P[i] ^ streamtoword(key, koffp);
611            }
612    
613            for (i = 0; i < plen; i += 2) {
614                lr[0] ^= streamtoword(data, doffp);
615                lr[1] ^= streamtoword(data, doffp);
616                encipher(lr, 0);
617                this.P[i] = lr[0];
618                this.P[i + 1] = lr[1];
619            }
620    
621            for (i = 0; i < slen; i += 2) {
622                lr[0] ^= streamtoword(data, doffp);
623                lr[1] ^= streamtoword(data, doffp);
624                encipher(lr, 0);
625                this.S[i] = lr[0];
626                this.S[i + 1] = lr[1];
627            }
628        }

630        /**
631         * Perform the central password hashing step in the
632         * bcrypt scheme
633         * @param password    the password to hash
634         * @param salt    the binary salt to hash with the password
635         * @param log_rounds    the binary logarithm of the number
636         * of rounds of hashing to apply
637         * @return    an array containing the binary hashed password
638         */
639        private byte[] crypt_raw(byte password[], byte salt[], int log_rounds) {
640            int rounds, i, j;
641            int cdata[] = bf_crypt_ciphertext.clone();
642            int clen = cdata.length;
643            byte ret[];
644    
645            if (log_rounds < 4 || log_rounds > 31) {
646                throw new IllegalArgumentException ("Bad number of rounds");
647            }
648            rounds = 1 << log_rounds;
649            if (salt.length != BCRYPT_SALT_LEN) {
650                throw new IllegalArgumentException ("Bad salt length");
651            }
652    
653            init_key();
654            ekskey(salt, password);
655            for (i = 0; i < rounds; i++) {
656                key(password);
657                key(salt);
658            }
659    
660            for (i = 0; i < 64; i++) {
661                for (j = 0; j < (clen >> 1); j++) {
662                    encipher(cdata, j << 1);
663                }
664            }
665    
666            ret = new byte[clen * 4];
667            for (i = 0, j = 0; i < clen; i++) {
668                ret[j++] = (byte)((cdata[i] >> 24) & 0xff);
669                ret[j++] = (byte)((cdata[i] >> 16) & 0xff);
670                ret[j++] = (byte)((cdata[i] >> 8) & 0xff);
671                ret[j++] = (byte)(cdata[i] & 0xff);
672            }
673            return ret;
674        }

676        /**
677         * Hash a password using the OpenBSD bcrypt scheme
678         * @param password    the password to hash
679         * @param salt    the salt to hash with (perhaps generated
680         * using BCrypt.gensalt)
681         * @return    the hashed password
682         */
683        public static String hashpw(String password, String salt) {
684            BCrypt B;
685            String real_salt;
686            byte passwordb[], saltb[], hashed[];
687            char minor = (char)0;
688            int rounds, off = 0;
689            StringBuffer rs = new StringBuffer();
690    
691            if (salt.charAt(0) != '$' || salt.charAt(1) != '2') {
692                throw new IllegalArgumentException ("Invalid salt version");
693            }
694            if (salt.charAt(2) == '$') {
695                off = 3;
696            } else {
697                minor = salt.charAt(2);
698                if (minor != 'a' || salt.charAt(3) != '$') {
699                    throw new IllegalArgumentException ("Invalid salt revision");
700                }
701                off = 4;
702            }
703    
704            // Extract number of rounds
705            if (salt.charAt(off + 2) > '$') {
706                throw new IllegalArgumentException ("Missing salt rounds");
707            }
708            rounds = Integer.parseInt(salt.substring(off, off + 2));
709    
710            real_salt = salt.substring(off + 3, off + 25);
711            try {
712                passwordb = (password + (minor >= 'a' ? "\000" : ""))
713                        .getBytes("UTF-8");
714            } catch (UnsupportedEncodingException uee) {
715                throw new UnsupportedOperationException("UTF-8 is not supported", uee);
716            }
717    
718            saltb = decode_base64(real_salt, BCRYPT_SALT_LEN);
719    
720            B = new BCrypt();
721            hashed = B.crypt_raw(passwordb, saltb, rounds);
722    
723            rs.append("$2");
724            if (minor >= 'a') {
725                rs.append(minor);
726            }
727            rs.append("$");
728            if (rounds < 10) {
729                rs.append("0");
730            }
731            rs.append(Integer.toString(rounds));
732            rs.append("$");
733            rs.append(encode_base64(saltb, saltb.length));
734            rs.append(encode_base64(hashed,
735                bf_crypt_ciphertext.length * 4 - 1));
736            return rs.toString();
737        }

740         * Generate a salt for use with the BCrypt.hashpw() method
741         * @param log_rounds    the log2 of the number of rounds of
742         * hashing to apply - the work factor therefore increases as
743         * 2**log_rounds.
744         * @param random        an instance of SecureRandom to use

757            rs.append(Integer.toString(log_rounds));
