117        private static long[][] C = new long[8][256];
118        private static long[]  rc = new long[R + 1];
119    
120        static {
121            for (int x = 0; x < 256; x++) {
122                char c = sbox.charAt(x/2);
123                long v1 = ((x & 1) == 0) ? c >>> 8 : c & 0xff;
124                long v2 = v1 << 1;
125                if (v2 >= 0x100L) {
126                    v2 ^= 0x11dL;
127                }
128                long v4 = v2 << 1;
129                if (v4 >= 0x100L) {
130                    v4 ^= 0x11dL;
131                }
132                long v5 = v4 ^ v1;
133                long v8 = v4 << 1;
134                if (v8 >= 0x100L) {
135                    v8 ^= 0x11dL;
136                }
137                long v9 = v8 ^ v1;
138                /*
139                 * build the circulant table C[0][x] = S[x].[1, 1, 4, 1, 8, 5, 2, 9]:
140                 */
141                C[0][x] =
142                    (v1 << 56) | (v1 << 48) | (v4 << 40) | (v1 << 32) |
143                    (v8 << 24) | (v5 << 16) | (v2 <<  8) | (v9      );
144                /*
145                 * build the remaining circulant tables C[t][x] = C[0][x] rotr t
146                 */
147                for (int t = 1; t < 8; t++) {
148                    C[t][x] = (C[t - 1][x] >>> 8) | ((C[t - 1][x] << 56));
149                }
150            }
151    
152            /*
153             * build the round constants:
154             */
155            rc[0] = 0L; /* not used (assigment kept only to properly initialize all variables) */
156            for (int r = 1; r <= R; r++) {
157                int i = 8*(r - 1);
158                rc[r] =
159                    (C[0][i    ] & 0xff00000000000000L) ^
160                    (C[1][i + 1] & 0x00ff000000000000L) ^
161                    (C[2][i + 2] & 0x0000ff0000000000L) ^
162                    (C[3][i + 3] & 0x000000ff00000000L) ^
163                    (C[4][i + 4] & 0x00000000ff000000L) ^
164                    (C[5][i + 5] & 0x0000000000ff0000L) ^
165                    (C[6][i + 6] & 0x000000000000ff00L) ^
166                    (C[7][i + 7] & 0x00000000000000ffL);
167            }
168        }

170        /**
171         * Global number of hashed bits (256-bit counter).

173        protected byte[] bitLength = new byte[32];
174    
175        /**
176         * Buffer of data to hash.
177         */
178        protected byte[] buffer = new byte[64];
179    
180        /**
181         * Current number of bits on the buffer.
182         */
183        protected int bufferBits = 0;
184    
185        /**
186         * Current (possibly incomplete) byte slot on the buffer.
187         */
188        protected int bufferPos = 0;
189    
190        /**
191         * The hashing state.
192         */
193        protected long[] hash  = new long[8];
194        protected long[] K     = new long[8]; // the round key
195        protected long[] L     = new long[8];
196        protected long[] block = new long[8]; // mu(buffer)
197        protected long[] state = new long[8]; // the cipher state
198    
199        public Whirlpool() {
200        }
201    
202        /**
203         * The core Whirlpool transform.
204         */
205        protected void processBuffer() {
206            /*
207             * map the buffer to a block:
208             */
209            for (int i = 0, j = 0; i < 8; i++, j += 8) {
210                block[i] =
211                    (((long)buffer[j    ]        ) << 56) ^
212                    (((long)buffer[j + 1] & 0xffL) << 48) ^
213                    (((long)buffer[j + 2] & 0xffL) << 40) ^
214                    (((long)buffer[j + 3] & 0xffL) << 32) ^
215                    (((long)buffer[j + 4] & 0xffL) << 24) ^
216                    (((long)buffer[j + 5] & 0xffL) << 16) ^
217                    (((long)buffer[j + 6] & 0xffL) <<  8) ^
218                    (((long)buffer[j + 7] & 0xffL)      );
219            }
220            /*
221             * compute and apply K^0 to the cipher state:
222             */
223            for (int i = 0; i < 8; i++) {
224                state[i] = block[i] ^ (K[i] = hash[i]);
225            }
226            /*
227             * iterate over all rounds:
228             */
229            for (int r = 1; r <= R; r++) {
230                /*
231                 * compute K^r from K^{r-1}:
232                 */
233                for (int i = 0; i < 8; i++) {
234                    L[i] = 0L;
235                    for (int t = 0, s = 56; t < 8; t++, s -= 8) {
236                        L[i] ^= C[t][(int)(K[(i - t) & 7] >>> s) & 0xff];
237                    }
238                }
239                for (int i = 0; i < 8; i++) {
240                    K[i] = L[i];
241                }
242                K[0] ^= rc[r];
243                /*
244                 * apply the r-th round transformation:
245                 */
246                for (int i = 0; i < 8; i++) {
247                    L[i] = K[i];
248                    for (int t = 0, s = 56; t < 8; t++, s -= 8) {
249                        L[i] ^= C[t][(int)(state[(i - t) & 7] >>> s) & 0xff];
250                    }
251                }
252                for (int i = 0; i < 8; i++) {
253                    state[i] = L[i];
254                }
255            }
256            /*
257             * apply the Miyaguchi-Preneel compression function:
258             */
259            for (int i = 0; i < 8; i++) {
260                hash[i] ^= state[i] ^ block[i];
261            }
262        }

264        /**
265         * Initialize the hashing state.
266         */
267        public void NESSIEinit() {
268            Arrays.fill(bitLength, (byte)0);
269            bufferBits = bufferPos = 0;
270            buffer[0] = 0; // it's only necessary to cleanup buffer[bufferPos].
271            Arrays.fill(hash, 0L); // initial value
272        }
273    
274        /**
275         * Delivers input data to the hashing algorithm.
276         *
277         * @param    source        plaintext data to hash.
278         * @param    sourceBits    how many bits of plaintext to process.
279         *
280         * This method maintains the invariant: bufferBits < 512
281         */
282        public void NESSIEadd(byte[] source, long sourceBits) {
283            /*
284                               sourcePos
285                               |
286                               +-------+-------+-------
287                                  ||||||||||||||||||||| source
288                               +-------+-------+-------
289            +-------+-------+-------+-------+-------+-------
290            ||||||||||||||||||||||                           buffer
291            +-------+-------+-------+-------+-------+-------
292                            |
293                            bufferPos
294            */
295            int sourcePos = 0; // index of leftmost source byte containing data (1 to 8 bits).
296            int sourceGap = (8 - ((int)sourceBits & 7)) & 7; // space on source[sourcePos].
297            int bufferRem = bufferBits & 7; // occupied bits on buffer[bufferPos].
298            int b;
299            // tally the length of the added data:
300            long value = sourceBits;
301            for (int i = 31, carry = 0; i >= 0; i--) {
302                carry += (bitLength[i] & 0xff) + ((int)value & 0xff);
303                bitLength[i] = (byte)carry;
304                carry >>>= 8;
305                value >>>= 8;
306            }
307    
308            // process data in chunks of 8 bits:
309            while (sourceBits > 8) { // at least source[sourcePos] and source[sourcePos+1] contain data.
310                // take a byte from the source:
311                b = ((source[sourcePos] << sourceGap) & 0xff) |
312                    ((source[sourcePos + 1] & 0xff) >>> (8 - sourceGap));
313                if (b < 0 || b >= 256) {
314                    throw new RuntimeException("LOGIC ERROR");
315                }
316                // process this byte:
317                buffer[bufferPos++] |= b >>> bufferRem;
318                bufferBits += 8 - bufferRem; // bufferBits = 8*bufferPos;
319                if (bufferBits == 512) {
320                    // process data block:
321                    processBuffer();
322                    // reset buffer:
323                    bufferBits = bufferPos = 0;
324                }
325                buffer[bufferPos] = (byte)((b << (8 - bufferRem)) & 0xff);
326                bufferBits += bufferRem;
327                // proceed to remaining data:
328                sourceBits -= 8;
329                sourcePos++;
330            }
331    
332            // now 0 <= sourceBits <= 8;
333            // furthermore, all data (if any is left) is in source[sourcePos].
334            if (sourceBits > 0) {
335                b = (source[sourcePos] << sourceGap) & 0xff; // bits are left-justified on b.
336                // process the remaining bits:
337                buffer[bufferPos] |= b >>> bufferRem;
338            } else {
339                b = 0;
340            }
341            if (bufferRem + sourceBits < 8) {
342                // all remaining data fits on buffer[bufferPos], and there still remains some space.
343                bufferBits += sourceBits;
344            } else {
345                // buffer[bufferPos] is full:
346                bufferPos++;
347                bufferBits += 8 - bufferRem; // bufferBits = 8*bufferPos;
348                sourceBits -= 8 - bufferRem;
349                // now 0 <= sourceBits < 8; furthermore, all data is in source[sourcePos].
350                if (bufferBits == 512) {
351                    // process data block:
352                    processBuffer();
353                    // reset buffer:
354                    bufferBits = bufferPos = 0;
355                }
356                buffer[bufferPos] = (byte)((b << (8 - bufferRem)) & 0xff);
357                bufferBits += (int)sourceBits;
358            }
359        }

361        /**
362         * Get the hash value from the hashing state.
363         *
364         * This method uses the invariant: bufferBits < 512
365         */
366        public void NESSIEfinalize(byte[] digest) {
367            // append a '1'-bit:
368            buffer[bufferPos] |= 0x80 >>> (bufferBits & 7);
369            bufferPos++; // all remaining bits on the current byte are set to zero.
370            // pad with zero bits to complete 512N + 256 bits:
371            if (bufferPos > 32) {
372                while (bufferPos < 64) {
373                    buffer[bufferPos++] = 0;
374                }
375                // process data block:
376                processBuffer();
377                // reset buffer:
378                bufferPos = 0;
379            }
380            while (bufferPos < 32) {
381                buffer[bufferPos++] = 0;
382            }
383            // append bit length of hashed data:
384            System.arraycopy(bitLength, 0, buffer, 32, 32);
385            // process data block:
386            processBuffer();
387            // return the completed message digest:
388            for (int i = 0, j = 0; i < 8; i++, j += 8) {
389                long h = hash[i];
390                digest[j    ] = (byte)(h >>> 56);
391                digest[j + 1] = (byte)(h >>> 48);
392                digest[j + 2] = (byte)(h >>> 40);
393                digest[j + 3] = (byte)(h >>> 32);
394                digest[j + 4] = (byte)(h >>> 24);
395                digest[j + 5] = (byte)(h >>> 16);
396                digest[j + 6] = (byte)(h >>>  8);
397                digest[j + 7] = (byte)(h       );
398            }
399        }

401        /**
402         * Delivers string input data to the hashing algorithm.
403         *
404         * @param    source        plaintext data to hash (ASCII text string).
405         *
406         * This method maintains the invariant: bufferBits < 512
407         */
408        public void NESSIEadd(String source) {
409            if (source.length() > 0) {
410                byte[] data = new byte[source.length()];
411                for (int i = 0; i < source.length(); i++) {
412                    data[i] = (byte)source.charAt(i);
413                }
414                NESSIEadd(data, 8*data.length);
415            }
416        }

418        public static String display(byte[] array) {
419            char[] val = new char[2*array.length];
420            //String hex = "0123456789ABCDEF";
421            String hex = "0123456789abcdef";
422            for (int i = 0; i < array.length; i++) {
423                int b = array[i] & 0xff;
424                val[2*i] = hex.charAt(b >>> 4);
425                val[2*i + 1] = hex.charAt(b & 15);
426            }
427            return String.valueOf(val);
428        }
