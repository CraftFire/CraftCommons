020    package com.craftfire.commons.ip;
021    
022    import java.net.Inet4Address;
023    import java.net.InetAddress;
024    import java.net.UnknownHostException;
025    import java.nio.ByteBuffer;
026    import java.util.Arrays;
027    
028    /**
029     * A class that represents an IPv4 address.
030     */
031    public class IPv4Address extends IPAddress {
032        private final byte[] data;
033    
034        /**
035         * Creates an IPv4Address object for given Inet4Address.
036         * 
037         * @param address  the Inet4Address
038         * @see            IPAddress#valueOf(InetAddress)
039         */
040        public IPv4Address(Inet4Address address) {
041            this.data = address.getAddress();
042        }
043    
044        /**
045         * Creates an IPv4Address object form given byte representation.
046         * 
047         * @param  address                   byte representation of the IPv4 address, <b>must be 4 byte long</b>
048         * @throws IllegalArgumentException  if the number of bytes is different than 4
049         */
050        public IPv4Address(byte... address) {
051            if (address.length != 4) {
052                throw new IllegalArgumentException("IPv4 Address must have 4 parts!");
053            }
054            this.data = address.clone();
055        }
056    
057        /**
058         * Creates an IPv4Address object from four integers corresponding to four parts of IPv4 address (usually seperated by dots).
059         * 
060         * @param  address                   integer parts of the IPv4 address, <b>must be exactly 4 elements</b>
061         * @throws IllegalArgumentException  if the number of ints is different than 4
062         */
063        public IPv4Address(int... address) {
064            if (address.length != 4) {
065                throw new IllegalArgumentException("IPv4 Address must have 4 parts!");
066            }
067            this.data = new byte[] { 0, 0, 0, 0 };
068            for (int i = 0; i < 4; ++i) {
069                this.data[i] = (byte) address[i];
070            }
071        }
072    
073        /* (non-Javadoc)
074         * @see com.craftfire.commons.ip.IPAddress#isIPv4()
075         */
076        @Override
077        public boolean isIPv4() {
078            return true;
079        }
080    
081        /* (non-Javadoc)
082         * @see com.craftfire.commons.ip.IPAddress#isIPv6()
083         */
084        @Override
085        public boolean isIPv6() {
086            return false;
087        }
088    
089        /* (non-Javadoc)
090         * @see com.craftfire.commons.ip.IPAddress#getInetAddress()
091         */
092        @Override
093        public Inet4Address getInetAddress() {
094            try {
095                InetAddress address = InetAddress.getByAddress(this.data);
096                if (address instanceof Inet4Address) {
097                    return (Inet4Address) address;
098                }
099            } catch (UnknownHostException ignore) {
100            }
101            return null;
102        }
103    
104        /* (non-Javadoc)
105         * @see com.craftfire.commons.ip.IPAddress#toIPv4()
106         */
107        @Override
108        public IPv4Address toIPv4() {
109            return this;
110        }
111    
112        /* (non-Javadoc)
113         * @see com.craftfire.commons.ip.IPAddress#toIPv6()
114         */
115        @Override
116        public IPv6Address toIPv6() {
117            ByteBuffer buffer = ByteBuffer.allocate(16);
118            InetAddress inet = getInetAddress();
119            if (inet.isAnyLocalAddress()) {
120                return new IPv6Address(0, 0, 0, 0, 0, 0, 0, 0);
121            } else if (inet.isLoopbackAddress()) {
122                return new IPv6Address(0, 0, 0, 0, 0, 0, 0, 1);
123            } else if (inet.isSiteLocalAddress()) {
124                buffer.putShort((short) 0xfe80);
125            } else {
126                buffer.putShort((short) 0x2002);
127            }
128            buffer.position(12);
129            buffer.put(this.data);
130            short[] shorts = new short[8];
131            buffer.flip();
132            buffer.asShortBuffer().get(shorts, 0, 8);
133            return new IPv6Address(shorts);
134        }
135    
136        /* (non-Javadoc)
137         * @see com.craftfire.commons.ip.IPAddress#getBytes()
138         */
139        @Override
140        public byte[] getBytes() {
141            return getAddress();
142        }
143    
144        /**
145         * Returns four bytes, one for each part of IPv4 address (usually separated by dots).
146         * <p>
147         * In this case it returns the same value as {@link #getBytes()}
148         * 
149         * @return  four bytes, one for each part of IPv4 address
150         */
151        public byte[] getAddress() {
152            return this.data.clone();
153        }
154    
155        /**
156         * Returns the value of this address in single integer, which contains four bytes of the address.
157         * 
158         * @return  single integer representation of this address
159         */
160        public int getInt() {
161            return (byteToUint(this.data[0]) << 24) | (byteToUint(this.data[1]) << 16) | (byteToUint(this.data[2]) << 8) | byteToUint(this.data[3]);
162        }
163    
164        /* (non-Javadoc)
165         * @see java.lang.Object#toString()
166         */
167        @Override
168        public String toString() {
169            StringBuilder builder = new StringBuilder();
170            for (int i = 0; i < 4; ++i) {
171                builder.append(this.data[i] + ((this.data[i] < 0) ? 256 : 0));
172                if (i < 3) {
173                    builder.append(".");
174                }
175            }
176            return builder.toString();
177        }
178    
179        /* (non-Javadoc)
180         * @see java.lang.Object#equals(java.lang.Object)
181         */
182        @Override
183        public boolean equals(Object obj) {
184            if (obj == this) {
185                return true;
186            }
187            if (obj instanceof IPv4Address) {
188                return Arrays.equals(this.data, ((IPv4Address) obj).getAddress());
189            }
190            return false;
191        }
192    
193        /* (non-Javadoc)
194         * @see java.lang.Object#hashCode()
195         */
196        @Override
197        public int hashCode() {
198            return Arrays.hashCode(this.data);
199        }
200    
201        private static int byteToUint(byte b) {
202            return b + ((b < 0) ? 256 : 0);
203        }
204    }
