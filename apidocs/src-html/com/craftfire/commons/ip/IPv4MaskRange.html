020    package com.craftfire.commons.ip;
021    
022    /**
023     * An IPRange implementation for IPv4 masks.
024     * <p>
025     * If for example the data is binary: <code>11000000 10101000 00100100 00000001</code> (192.168.36.1)<br>
026     * And the bitmask is binary: <code>11111111 11111111 11111100 00000011</code> (255.255.252.3)
027     * <p>
028     * <code>data: 11000000 10101000 00100100 00000001<br>
029     * mask: 11111111 11111111 11111100 00000011<br>
030     * ----: 11000000 10101000 001001xx xxxxxx01</code>
031     * <p>
032     * A checked address must match 11000000 10101000 001001xx xxxxxx01 for {@link #isInRange(IPAddress)} to return true.
033     * 
034     */
035    public class IPv4MaskRange implements IPRange {
036        private final IPv4Address data, mask;
037    
038        /**
039         * Creates an IPv4MaskRange object with given {@link data} and {@link mask}.
040         * 
041         * @param data  an address that is compared to a checked address with given bitmask
042         * @param mask  a bitmask - binary 0 means that the bit can be any, 1 means that the bit in a checked address must equal to corrseponding bit in {@link data}
043         */
044        public IPv4MaskRange(IPv4Address data, IPv4Address mask) {
045            this.data = data;
046            this.mask = mask;
047        }
048    
049        /**
050         * Returns the address that is compared to a checked address with the bitmask.
051         * 
052         * @return the address
053         */
054        public IPv4Address getData() {
055            return this.data;
056        }
057    
058        /**
059         * Returns the bitmask that is used to compare the data with a checked address.
060         * 
061         * @return the bitmask
062         */
063        public IPv4Address getMask() {
064            return this.mask;
065        }
066    
067        /**
068         * Returns true, if the given IP address is an IPv4 address and matches the bitmask.
069         */
070        @Override
071        public boolean isInRange(IPAddress address) {
072            if (!address.isIPv4()) {
073                return false;
074            }
075            return ((address.toIPv4().getInt() ^ getData().getInt()) & getMask().getInt()) == 0;
076        }
077    
078        /* (non-Javadoc)
079         * @see java.lang.Object#toString()
080         */
081        @Override
082        public String toString() {
083            StringBuilder builder = new StringBuilder();
084            String maskStr = Integer.toBinaryString(getMask().getInt());
085            String dataStr = Integer.toBinaryString(getData().getInt());
086            for (int i = 0; i < 32; ++i) {
087                if (maskStr.charAt(i) == '0') {
088                    builder.append("x");
089                } else {
090                    builder.append(dataStr.charAt(i));
091                }
092                if (i % 8 == 7) {
093                    builder.append(" ");
094                }
095            }
096            builder.append("b");
097            return builder.toString();
098        }
099    }
