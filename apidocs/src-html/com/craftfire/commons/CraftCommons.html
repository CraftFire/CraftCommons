062        public static String long2ip(Long i) {
063            return ((i >> 24) & 0xFF) + "." +
064                   ((i >> 16) & 0xFF) + "." +
065                   ((i >> 8) & 0xFF) + "." +
066                   (i & 0xFF);
067        }

069        public static Long ip2long(String ip) {
070            String[] ipArray = ip.split("\\.");
071            long num = 0;
072            for (int i = 0; i < ipArray.length; i++) {
073                int power = 3 - i;
074                num += ((Integer.parseInt(ipArray[i]) % 256 * Math.pow(256, power)));
075            }
076            return num;
077        }

106        public static String encrypt(Encryption encryption, Object object, String salt, int iterationCount) {
107            try {
108                String string = (String) object;
109                MessageDigest md = null;
110                String newSalt = salt;
111                if (encryption.equals(Encryption.MD5)) {
112                    md = MessageDigest.getInstance("MD5");
113                } else if (encryption.equals(Encryption.SHA1)) {
114                    md = MessageDigest.getInstance("SHA-1");
115                } else if (encryption.equals(Encryption.SHA256)) {
116                    md = MessageDigest.getInstance("SHA-256");
117                } else if (encryption.equals(Encryption.SHA512)) {
118                    md = MessageDigest.getInstance("SHA-512");
119                } else if (encryption.equals(Encryption.CRC32)) {
120                    Checksum checksum = new CRC32();
121                    checksum.update(string.getBytes(), 0, string.getBytes().length);
122                    return Long.toString(checksum.getValue());
123                } else if (encryption.equals(Encryption.WHIRLPOOL)) {
124                    Whirlpool w = new Whirlpool();
125                    byte[] digest = new byte[Whirlpool.DIGESTBYTES];
126                    w.NESSIEinit();
127                    w.NESSIEadd(string);
128                    w.NESSIEfinalize(digest);
129                    return Whirlpool.display(digest);
130                } else if (encryption.equals(Encryption.PHPASS_P) || encryption.equals(Encryption.PHPASS_H)) {
131                    PHPass phpass;
132                    if (iterationCount == 0) {
133                        phpass = new PHPass(encryption, 8);
134                    } else {
135                        phpass = new PHPass(encryption, iterationCount);
136                    }
137                    if (newSalt == null || newSalt.isEmpty()) {
138                        newSalt = phpass.gensalt();
139                    }
140                    String hash = phpass.crypt(string, newSalt);
141                    if (hash.length() == 34) {
142                        return hash;
143                    }
144                    return "*";
145                } else if (encryption.equals(Encryption.BLOWFISH)) {
146                    if (newSalt == null || newSalt.isEmpty()) {
147                        if (iterationCount == 0) {
148                            newSalt = BCrypt.gensalt(8);
149                        } else {
150                            newSalt = BCrypt.gensalt(iterationCount);
151                        }
152                    }
153                    return BCrypt.hashpw(string, newSalt);
154                }
155                if (md != null) {
156                    md.update(string.getBytes("ISO-8859-1"), 0, string.length());
157                    byte[] hash = md.digest();
158                    return EncryptionUtil.bytesTohex(hash);
159                } else {
160                    return null;
161                }
162            } catch (GeneralSecurityException e) {
163                throw new IllegalStateException(e);
164            } catch (UnsupportedEncodingException e) {
165                throw new UnsupportedOperationException(e);
166            }
167        }

169        public static Encryption unixHashIdentify(String hash){
170            if (hash.startsWith("$1$") || hash.startsWith("$md5$")) {
171                return Encryption.MD5;
172            } else if (hash.startsWith("$2")) {
173                return Encryption.BLOWFISH;
174            } else if (hash.startsWith("$5$")) {
175                return Encryption.SHA256;
176            } else if (hash.startsWith("$6$")) {
177                return Encryption.SHA512;
178            } else if (hash.startsWith("$P$")) {
179                return Encryption.PHPASS_P;
180            } else if (hash.startsWith("$H$")) {
181                return Encryption.PHPASS_H;
182            }
183            return null;
184        }

223        public static String convertHexToString(String hex) {
224            StringBuilder sb = new StringBuilder();
225            for ( int i=0; i<hex.length()-1; i+=2 ) {
226                String output = hex.substring(i, (i + 2));
227                int decimal = Integer.parseInt(output, 16);
228                sb.append((char)decimal);
229            }
230            return sb.toString();
231        }

233        public static String convertStringToHex(String str) {
234            char[] chars = str.toCharArray();
235            StringBuffer hex = new StringBuffer();
236            for (int i = 0; i < chars.length; i++){
237                hex.append(Integer.toHexString(chars[i]));
238            }
239            return hex.toString();
240        }

335        public static String removeChar(String s, char c) {
336            StringBuilder r = new StringBuilder(s.length());
337            r.setLength(s.length());
338            int current = 0;
339            for (int i = 0; i < s.length(); i++) {
340                char cur = s.charAt(i);
341                if (cur != c) {
342                    r.setCharAt(current++, cur);
343                }
344            }
345            return r.toString();
346        }
