020    package com.craftfire.commons.analytics;
021    
022    import java.util.HashMap;
023    import java.util.Map;
024    
025    /**
026     * This class contains all the reporting data for the {@link AnalyticsManager}.
027     *
028     * @see AnalyticsManager
029     * @see AnalyticsItem
030     */
031    public class AnalyticsData {
032        private AnalyticsItem name, version;
033    
034        /**
035         * Contains all the custom data for the analytics.
036         */
037        private Map<String, AnalyticsItem> customData = new HashMap<String, AnalyticsItem>();
038    
039        /**
040         * Default constructor for the class.
041         *
042         * @param name     the name of the item that is being reported
043         * @param version  the version of the item that is being reported
044         */
045        public AnalyticsData(String name, String version) {
046            this.name = new AnalyticsItem("project_name", name);
047            this.version = new AnalyticsItem("project_version",version);
048        }
049    
050        /**
051         * Returns the name of the data item.
052         *
053         * @return name of the data item
054         */
055        public AnalyticsItem getName() {
056            return this.name;
057        }
058    
059        /**
060         * Sets the name of the data item.
061         *
062         * @param name  name of the data item
063         */
064        public void setName(String name) {
065            getName().setValue(name);
066        }
067    
068        /**
069         * Returns the version of the data item.
070         *
071         * @return version of the data item
072         */
073        public AnalyticsItem getVersion() {
074            return this.version;
075        }
076    
077        /**
078         * Sets the version of the data item.
079         *
080         * @param version  version of the data item
081         */
082        public void setVersion(String version) {
083            getVersion().setValue(version);
084        }
085    
086        /**
087         * Adds a custom data item.
088         * <p>
089         * Use this method if you wish to add custom values to report.
090         *
091         * @param key    the key of the custom item
092         * @param value  the value of the custom item
093         */
094        public void add(String key, String value) {
095            this.customData.put(key.toLowerCase(), new AnalyticsItem(key.toLowerCase(), value));
096        }
097    
098        /**
099         * Returns the custom data items.
100         *
101         * @return custom data items
102         */
103        public Map<String, AnalyticsItem> getCustomData() {
104            return this.customData;
105        }
106    
107        /**
108         * Returns a specific data item if {@code key} exists, returns {@code null} if the {@code key} was not found.
109         *
110         * @param  key  key identifier of the custom data item
111         * @return      custom data item if found, null if {@code key} was not found
112         */
113        public AnalyticsItem getCustom(String key) {
114            if (getCustomData().containsKey(key.toLowerCase())) {
115                return getCustomData().get(key.toLowerCase());
116            }
117            return null;
118        }
119    
120        /**
121         * Returns the operating system name.
122         * <pre><b>Example:</b> Linux</pre>
123         *
124         * @return operating system name
125         */
126        public AnalyticsItem getOSName() {
127            return new AnalyticsItem("os_name", System.getProperty("os.name"));
128        }
129    
130        /**
131         * Returns the operating system version.
132         * <pre><b>Example:</b> 3.2.0-3-amd64</pre>
133         *
134         * @return operating system version
135         */
136        public AnalyticsItem getOSVersion() {
137            return new AnalyticsItem("os_version", System.getProperty("os.version"));
138        }
139    
140        /**
141         * Returns the operating system architecture.
142         * <pre><b>Example:</b> amd64</pre>
143         *
144         * @return operating system architecture
145         */
146        public AnalyticsItem getOSArch() {
147            return new AnalyticsItem("os_arch", System.getProperty("os.arch"));
148        }
149    
150        /**
151         * Returns the amount of memory used.
152         * <pre><b>Example:</b> 2813722624</pre>
153         *
154         * @return operating system name
155         */
156        public AnalyticsItem getTotalMemory() {
157            return new AnalyticsItem("ram_total", Long.toString(Runtime.getRuntime().totalMemory()));
158        }
159    
160        /**
161         * Returns the total amount of memory.
162         * <pre><b>Example:</b> 2813722624</pre>
163         *
164         * @return operating system name
165         */
166        public AnalyticsItem getMaxMemory() {
167            return new AnalyticsItem("ram_max", Long.toString(Runtime.getRuntime().maxMemory()));
168        }
169    
170        /**
171         * Returns the Java version.
172         * <pre><b>Example:</b> 1.6.0_24</pre>
173         *
174         * @return java version
175         */
176        public AnalyticsItem getJavaVersion() {
177            return new AnalyticsItem("java_version", System.getProperty("java.version"));
178        }
179    }
